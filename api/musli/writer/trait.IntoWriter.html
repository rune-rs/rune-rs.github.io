Expand description

Coerce a type into a Writer.

§Examples

use musli::{Context, IntoWriter, Writer};
use musli::context;

let mut buffer = Vec::new();
let mut writer = (&mut buffer).into_writer();
let cx = context::new();

writer.write_bytes(&cx, b"Hello")?;
writer.finish(&cx)?;

assert_eq!(buffer, b"Hello");
