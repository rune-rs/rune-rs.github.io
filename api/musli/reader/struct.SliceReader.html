§Examples
use musli::Context;
use musli::reader::{Reader, SliceReader};

fn process<C>(cx: &C) -> Result<(), C::Error>
where
    C: ?Sized + Context
{
    let mut reader = SliceReader::new(&[1, 2, 3, 4]);
    assert_eq!(reader.as_slice(), &[1, 2, 3, 4]);
    reader.skip(cx, 2)?;
    assert_eq!(reader.as_slice(), &[3, 4]);
    Ok(())
}
