33/// Derive to implement the `TryClone` trait.
34///
35/// # Examples
36///
37/// Basic usage example:
38///
39/// ```
40/// use rune::alloc::String;
41/// use rune::alloc::clone::TryClone;
42///
43/// // String type implements TryClone
44/// let s = String::new();
45/// // ... so we can clone it
46/// let copy = s.try_clone()?;
47/// # Ok::<_, rune::alloc::Error>(())
48/// ```
49///
50/// To easily implement the TryClone trait, you can also use
51/// `#[derive(TryClone)]`. Example:
52///
53/// ```
54/// use rune::alloc::clone::TryClone;
55///
56/// // we add the TryClone trait to Morpheus struct
57/// #[derive(TryClone)]
58/// struct Morpheus {
59///    blue_pill: f32,
60///    red_pill: i64,
61/// }
62///
63/// let f = Morpheus { blue_pill: 0.0, red_pill: 0 };
64/// // and now we can clone it!
65/// let copy = f.try_clone()?;
66/// # Ok::<_, rune::alloc::Error>(())
67/// ```
68///
69/// ## Attributes
70///
71/// ### `try_clone(with = <path>)`
72///
73/// Specify a custom method when cloning a field.
74///
75/// ```
76/// use rune::alloc::clone::TryClone;
77///
78/// #[derive(Debug, TryClone)]
79/// #[non_exhaustive]
80/// pub struct Struct {
81///     #[try_clone(with = String::clone)]
82///     string: String,
83/// }
84/// ```
85///
86/// ### `try_clone(try_with = <path>)`
87///
88/// Specify a custom fallible method when cloning a field.
89///
90/// ```
91/// use rune::alloc::clone::TryClone;
92///
93/// #[derive(Debug, TryClone)]
94/// #[non_exhaustive]
95/// pub struct Struct {
96///     #[try_clone(try_with = rune::alloc::String::try_clone)]
97///     string: rune::alloc::String,
98/// }
99/// ```
100///
101/// ### `try_clone(copy)`
102///
103/// Specify that a field is `Copy`.
104///
105/// ```
106/// use rune::alloc::prelude::*;
107///
108/// #[derive(Debug, TryClone)]
109/// #[non_exhaustive]
110/// pub struct Struct {
111///     #[try_clone(copy)]
112///     number: u32,
113/// }
114/// ```
115#[proc_macro_derive(TryClone, attributes(try_clone))]
116pub fn try_clone(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
117    let input = syn::parse_macro_input!(input as syn::DeriveInput);
118
119    try_clone::expand(input)
120        .unwrap_or_else(to_compile_errors)
121        .into()
122}
123
124fn to_compile_errors<I>(errors: I) -> proc_macro2::TokenStream
125where
126    I: IntoIterator<Item = syn::Error>,
127{
128    let compile_errors = errors.into_iter().map(syn::Error::into_compile_error);
129    ::quote::quote!(#(#compile_errors)*)
130}
