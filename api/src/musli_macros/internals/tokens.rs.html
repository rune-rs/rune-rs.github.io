77fn path<const N: usize>(span: Span, prefix: &syn::Path, segments: [&'static str; N]) -> syn::Path {
78    let mut path = prefix.clone();
79
80    for segment in segments {
81        path.segments.push(syn::Ident::new(segment, span).into());
82    }
83
84    path
