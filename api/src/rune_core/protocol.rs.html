10/// A pre-defined protocol function.
11///
12/// # Examples
13///
14/// ```
15/// use rune::runtime::Protocol;
16///
17/// let protocol = Protocol::GET;
18/// assert_eq!(protocol.name, "GET");
19/// ```
20#[derive(Debug)]
21#[non_exhaustive]
22pub struct Protocol {
23    /// The name of the builtin function.
24    pub name: &'static str,
25    /// If this protocol defines an associated method, this is the name of that
26    /// method.
27    pub method: Option<&'static str>,
28    /// The hash of the builtin function.
29    pub hash: Hash,
30    /// Representative expression for the protocol.
31    ///
32    /// If no such expression is present, then it means that its an internal
33    /// protocol.
34    #[cfg(feature = "doc")]
35    #[doc(hidden)]
36    pub repr: Option<&'static str>,
37    /// Documentation for protocol.
38    #[cfg(feature = "doc")]
39    #[doc(hidden)]
40    pub doc: &'static [&'static str],
41}
42
43impl ToTypeHash for &Protocol {
44    #[inline]
45    fn to_type_hash(&self) -> Hash {
46        self.hash
47    }
48
49    #[inline]
50    #[cfg(feature = "alloc")]
51    fn to_item(&self) -> alloc::Result<Option<ItemBuf>> {
52        Ok(None)
53    }
54}
55
56impl fmt::Display for Protocol {
57    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
58        write!(f, "{}", self.name)
59    }
60}
61
62impl cmp::PartialEq for Protocol {
63    #[inline]
64    fn eq(&self, other: &Self) -> bool {
65        self.hash.eq(&other.hash)
66    }
67}
68
69impl cmp::Eq for Protocol {}
70
71impl core::hash::Hash for Protocol {
72    #[inline]
73    fn hash<H: Hasher>(&self, state: &mut H) {
74        self.hash.hash(state)
75    }
76}

78macro_rules! maybe {
79    ($tt:tt $($rest:tt)*) => { Some($tt $($rest)*) };
80    () => { None };
81}

83macro_rules! define {
84    (
85        $(
86            $(#[doc = $outer_doc:literal])*
87            $vis:vis const $ident:ident: Protocol = Protocol {
88                $(method: $method:expr,)?
89                hash: $hash:expr,
90                $(repr: $repr:expr,)?
91                $(#[doc = $doc:expr])*
92            };
93        )*
94    ) => {
95        impl Protocol {
96            $(
97                $(#[doc = $outer_doc])*
98                ///
99                /// # Examples
100                ///
101                /// ```
102                /// use rune::hash::IntoHash;
103                /// use rune::runtime::Protocol;
104                ///
105                #[doc = concat!(" let hash = Protocol::", stringify!($ident), ".into_hash();")]
106                /// let protocol = Protocol::from_hash(hash);
107                #[doc = concat!(" assert_eq!(protocol, Some(Protocol::", stringify!($ident), "));")]
108                /// ```
109                $vis const $ident: Protocol = Protocol {
110                    name: stringify!($ident),
111                    method: maybe!($($method)*),
112                    hash: Hash($hash),
113                    #[cfg(feature = "doc")]
114                    repr: maybe!($($repr)*),
115                    #[cfg(feature = "doc")]
116                    doc: &[$($doc),*],
117                };
118            )*
119
120            /// Look up protocol for the given hash.
121            ///
122            /// # Examples
123            ///
124            /// ```
125            /// use rune::runtime::Protocol;
126            ///
127            /// let hash = Protocol::GET.hash;
128            /// let protocol = Protocol::from_hash(hash).ok_or("missing protocol")?;
129            /// assert_eq!(protocol, Protocol::GET);
130            /// # Ok::<_, &'static str>(())
131            /// ```
132            pub const fn from_hash(hash: Hash) -> Option<Self> {
133                match hash {
134                    $(
135                        Hash($hash) => {
136                            Some(Self::$ident)
137                        },
138                    )*
139                    _ => None,
140                }
141            }
142        }
143
144        #[test]
145        fn ensure_unique_hashes() {
146            let mut map = ::rust_std::collections::HashMap::<Hash, &'static str>::new();
147
148            $(
149                if let Some(ident) = map.insert(Hash($hash), stringify!($ident)) {
150                    panic!("Trying to define protocol hash `{}` for `{}`, but it's already defined for {ident}", $hash, stringify!($ident));
151                }
152            )*
153        }
154    }
155}
