203/// Removes and returns the byte at position `index` within the Bytes,
204/// shifting all bytes after it to the left.
205///
206/// # Panics
207///
208/// Panics if `index` is out of bounds.
209///
210/// ```rune,should_panic
211/// let bytes = b"abc";
212/// bytes.remove(3);
213/// ```
214///
215/// # Examples
216///
217/// ```rune
218/// let bytes = b"abc";
219/// assert_eq!(bytes.remove(1), b'b');
220/// assert_eq!(bytes, b"ac");
221/// ```
222#[rune::function(instance)]
223fn remove(this: &mut Bytes, index: usize) -> VmResult<u8> {
224    if index >= this.len() {
225        return VmResult::err(VmErrorKind::OutOfRange {
226            index: index.into(),
227            length: this.len().into(),
228        });
229    }
230
231    let value = this.remove(index);
232    VmResult::Ok(value)
233}

235/// Inserts a byte at position `index` within the inner vector, shifting all
236/// elements after it to the right.
237///
238/// # Panics
239///
240/// Panics if `index` is out of bounds.
241///
242/// # Examples
243///
244/// ```rune
245/// let bytes = b"abc";
246/// bytes.insert(1, b'e');
247/// assert_eq!(bytes, b"aebc");
248/// bytes.insert(4, b'd');
249/// assert_eq!(bytes, b"aebcd");
250/// ```
251#[rune::function(instance)]
252fn insert(this: &mut Bytes, index: usize, value: u8) -> VmResult<()> {
253    if index > this.len() {
254        return VmResult::err(VmErrorKind::OutOfRange {
255            index: index.into(),
256            length: this.len().into(),
257        });
258    }
259
260    vm_try!(this.insert(index, value));
261    VmResult::Ok(())
262}

353/// Reserves capacity for at least `additional` more elements to be inserted in
354/// the given `Bytes`. The collection may reserve more space to speculatively
355/// avoid frequent reallocations. After calling `reserve`, capacity will be
356/// greater than or equal to `self.len() + additional`. Does nothing if capacity
357/// is already sufficient.
358///
359/// # Panics
360///
361/// Panics if the new capacity exceeds `isize::MAX` bytes.
362///
363/// # Examples
364///
365/// ```rune
366/// let vec = b"a";
367/// vec.reserve(10);
368/// assert!(vec.capacity() >= 11);
369/// ```
370#[rune::function(instance)]
371fn reserve(this: &mut Bytes, additional: usize) -> VmResult<()> {
372    vm_try!(this.reserve(additional));
373    VmResult::Ok(())
374}

376/// Reserves the minimum capacity for at least `additional` more elements to be
377/// inserted in the given `Bytes`. Unlike [`reserve`], this will not
378/// deliberately over-allocate to speculatively avoid frequent allocations.
379/// After calling `reserve_exact`, capacity will be greater than or equal to
380/// `self.len() + additional`. Does nothing if the capacity is already
381/// sufficient.
382///
383/// Note that the allocator may give the collection more space than it requests.
384/// Therefore, capacity can not be relied upon to be precisely minimal. Prefer
385/// [`reserve`] if future insertions are expected.
386///
387/// [`reserve`]: Bytes::reserve
388///
389/// # Panics
390///
391/// Panics if the new capacity exceeds `isize::MAX` bytes.
392///
393/// # Examples
394///
395/// ```rune
396/// let vec = b"a";
397/// vec.reserve_exact(10);
398/// assert!(vec.capacity() >= 11);
399/// ```
400#[rune::function(instance)]
401fn reserve_exact(this: &mut Bytes, additional: usize) -> VmResult<()> {
402    vm_try!(this.reserve_exact(additional));
403    VmResult::Ok(())
404}

552/// Returns a subslice of Bytes.
553///
554/// - If given a position, returns the byte at that position.
555/// - If given a range, returns the subslice corresponding to that range.
556///
557/// # Panics
558///
559/// Panics if `index` is out of bounds.
560///
561/// ```rune,should_panic
562/// let bytes = b"abc";
563/// assert_eq!(None, bytes[1..4]);
564/// ```
565///
566/// ```rune,should_panic
567/// let bytes = b"abc";
568/// assert_eq!(None, bytes[3]);
569/// ```
570///
571/// # Examples
572///
573/// ```rune
574/// let bytes = b"abcd";
575/// assert_eq!(bytes[0..2], b"ab");
576/// assert_eq!(bytes[0], b'a');
577/// ```
578#[rune::function(instance, protocol = INDEX_GET)]
579fn index_get(this: &Bytes, index: Value) -> VmResult<Value> {
580    match vm_try!(this.index_get(index)) {
581        Some(bytes) => VmResult::Ok(bytes),
582        None => VmResult::err(Panic::custom("missing bytes slice")),
583    }
