84    t.docs(docstring! {
85        /// Trait for comparisons using the equality operator.
86        ///
87        /// Implementing this trait for types provides the `==` and `!=`
88        /// operators for those types.
89        ///
90        /// `x.eq(y)` can also be written `x == y`, and `x.ne(y)` can be written
91        /// `x != y`. We use the easier-to-read infix notation in the remainder
92        /// of this documentation.
93        ///
94        /// This trait allows for comparisons using the equality operator, for
95        /// types that do not have a full equivalence relation. For example, in
96        /// floating point numbers `NaN != NaN`, so floating point types
97        /// implement `PartialEq` but not [`trait@Eq`]. Formally speaking, when
98        /// `Rhs == Self`, this trait corresponds to a [partial equivalence
99        /// relation].
100        ///
101        /// [partial equivalence relation]:
102        ///     https://en.wikipedia.org/wiki/Partial_equivalence_relation
103        ///
104        /// Implementations must ensure that `eq` and `ne` are consistent with
105        /// each other:
106        ///
107        /// - `a != b` if and only if `!(a == b)`.
108        ///
109        /// The default implementation of `ne` provides this consistency and is
110        /// almost always sufficient. It should not be overridden without very
111        /// good reason.
112        ///
113        /// If [`PartialOrd`] or [`Ord`] are also implemented for `Self` and
114        /// `Rhs`, their methods must also be consistent with `PartialEq` (see
115        /// the documentation of those traits for the exact requirements). It's
116        /// easy to accidentally make them disagree by deriving some of the
117        /// traits and manually implementing others.
118        ///
119        /// The equality relation `==` must satisfy the following conditions
120        /// (for all `a`, `b`, `c` of type `A`, `B`, `C`):
121        ///
122        /// - **Symmetry**: if `A: PartialEq<B>` and `B: PartialEq<A>`, then
123        ///   **`a == b` implies `b == a`**; and
124        ///
125        /// - **Transitivity**: if `A: PartialEq<B>` and `B: PartialEq<C>` and
126        ///   `A: PartialEq<C>`, then **`a == b` and `b == c` implies `a ==
127        ///   c`**. This must also work for longer chains, such as when `A:
128        ///   PartialEq<B>`, `B: PartialEq<C>`, `C: PartialEq<D>`, and `A:
129        ///   PartialEq<D>` all exist.
130        ///
131        /// Note that the `B: PartialEq<A>` (symmetric) and `A: PartialEq<C>`
132        /// (transitive) impls are not forced to exist, but these requirements
133        /// apply whenever they do exist.
134        ///
135        /// Violating these requirements is a logic error. The behavior
136        /// resulting from a logic error is not specified, but users of the
137        /// trait must ensure that such logic errors do *not* result in
138        /// undefined behavior. This means that `unsafe` code **must not** rely
139        /// on the correctness of these methods.
140        ///
141        /// ## Cross-crate considerations
142        ///
143        /// Upholding the requirements stated above can become tricky when one
144        /// crate implements `PartialEq` for a type of another crate (i.e., to
145        /// allow comparing one of its own types with a type from the standard
146        /// library). The recommendation is to never implement this trait for a
147        /// foreign type. In other words, such a crate should do `impl
148        /// PartialEq<ForeignType> for LocalType`, but it should *not* do `impl
149        /// PartialEq<LocalType> for ForeignType`.
150        ///
151        /// This avoids the problem of transitive chains that criss-cross crate
152        /// boundaries: for all local types `T`, you may assume that no other
153        /// crate will add `impl`s that allow comparing `T == U`. In other
154        /// words, if other crates add `impl`s that allow building longer
155        /// transitive chains `U1 == ... == T == V1 == ...`, then all the types
156        /// that appear to the right of `T` must be types that the crate
157        /// defining `T` already knows about. This rules out transitive chains
158        /// where downstream crates can add new `impl`s that "stitch together"
159        /// comparisons of foreign types in ways that violate transitivity.
160        ///
161        /// Not having such foreign `impl`s also avoids forward compatibility
162        /// issues where one crate adding more `PartialEq` implementations can
163        /// cause build failures in downstream crates.
164        ///
165        /// # Examples
166        ///
167        /// ```rune
168        /// let x = 0;
169        /// let y = 1;
170        ///
171        /// assert_eq!(x == y, false);
172        /// assert_eq!(x.eq(y), false);
173        ///
174        /// assert!((1.0).eq(1.0));
175        /// assert!(!(1.0).eq(2.0));
176        ///
177        /// assert!(1.0 == 1.0);
178        /// assert!(1.0 != 2.0);
179        /// ```
180    })?;
181
182    t.handler(|cx| {
183        let partial_eq = cx.find(&Protocol::PARTIAL_EQ)?;
184        let partial_eq = Caller::<(Value, Value), 2, bool>::new(partial_eq);
185
186        cx.function("ne", move |a: Value, b: Value| {
187            VmResult::Ok(!vm_try!(partial_eq.call((a, b))))
188        })?;
189
190        Ok(())
191    })?;

226        /// Trait for comparisons corresponding to [equivalence relations](
227        /// https://en.wikipedia.org/wiki/Equivalence_relation).
228        ///
229        /// This means, that in addition to `a == b` and `a != b` being strict
230        /// inverses, the relation must be (for all `a`, `b` and `c`):
231        ///
232        /// - reflexive: `a == a`;
233        /// - symmetric: `a == b` implies `b == a` (required by `PartialEq` as

235        /// - transitive: `a == b` and `b == c` implies `a == c` (required by
236        ///   `PartialEq` as well).
237        ///
238        /// This property cannot be checked by the compiler, and therefore `Eq`
239        /// implies [`PartialEq`], and has no extra methods.
240        ///
241        /// Violating this property is a logic error. The behavior resulting
242        /// from a logic error is not specified, but users of the trait must
243        /// ensure that such logic errors do *not* result in undefined behavior.
244        /// This means that `unsafe` code **must not** rely on the correctness
245        /// of these methods.
246        ///
247        /// Implement `Eq` in addition to `PartialEq` if it's guaranteed that
248        /// `PartialEq::eq(a, a)` always returns `true` (reflexivity), in
249        /// addition to the symmetric and transitive properties already required
250        /// by `PartialEq`.
251        /// ```

276    t.docs(docstring! {
277        /// Trait for types that form a [partial
278        /// order](https://en.wikipedia.org/wiki/Partial_order).
279        ///
280        /// The `lt`, `le`, `gt`, and `ge` methods of this trait can be called
281        /// using the `<`, `<=`, `>`, and `>=` operators, respectively.
282        ///
283        /// The methods of this trait must be consistent with each other and
284        /// with those of [`PartialEq`]. The following conditions must hold:
285        ///
286        /// 1. `a == b` if and only if `partial_cmp(a, b) == Some(Equal)`.
287        /// 2. `a < b` if and only if `partial_cmp(a, b) == Some(Less)`
288        /// 3. `a > b` if and only if `partial_cmp(a, b) == Some(Greater)`
289        /// 4. `a <= b` if and only if `a < b || a == b` 5. `a >= b` if and only
290        /// if `a > b || a == b`
291        /// 6. `a != b` if and only if `!(a == b)`.
292        ///
293        /// Conditions 2â€“5 above are ensured by the default implementation.
294        /// Condition 6 is already ensured by [`PartialEq`].
295        ///
296        /// If [`Ord`] is also implemented for `Self` and `Rhs`, it must also be
297        /// consistent with `partial_cmp` (see the documentation of that trait
298        /// for the exact requirements). It's easy to accidentally make them
299        /// disagree by deriving some of the traits and manually implementing
300        /// others.
301        ///
302        /// The comparison relations must satisfy the following conditions (for
303        /// all `a`, `b`, `c` of type `A`, `B`, `C`):
304        ///
305        /// - **Transitivity**: if `A: PartialOrd<B>` and `B: PartialOrd<C>` and
306        ///   `A: PartialOrd<C>`, then `a < b` and `b < c` implies `a < c`. The
307        ///   same must hold for both `==` and `>`. This must also work for
308        ///   longer chains, such as when `A: PartialOrd<B>`, `B:
309        ///   PartialOrd<C>`, `C: PartialOrd<D>`, and `A: PartialOrd<D>` all
310        ///   exist.
311        /// - **Duality**: if `A: PartialOrd<B>` and `B: PartialOrd<A>`, then `a
312        ///   < b` if and only if `b > a`.
313        ///
314        /// Note that the `B: PartialOrd<A>` (dual) and `A: PartialOrd<C>`
315        /// (transitive) impls are not forced to exist, but these requirements
316        /// apply whenever they do exist.
317        ///
318        /// Violating these requirements is a logic error. The behavior
319        /// resulting from a logic error is not specified, but users of the
320        /// trait must ensure that such logic errors do *not* result in
321        /// undefined behavior. This means that `unsafe` code **must not** rely
322        /// on the correctness of these methods.
323        ///
324        /// ## Cross-crate considerations
325        ///
326        /// Upholding the requirements stated above can become tricky when one
327        /// crate implements `PartialOrd` for a type of another crate (i.e., to
328        /// allow comparing one of its own types with a type from the standard
329        /// library). The recommendation is to never implement this trait for a
330        /// foreign type. In other words, such a crate should do `impl
331        /// PartialOrd<ForeignType> for LocalType`, but it should *not* do `impl
332        /// PartialOrd<LocalType> for ForeignType`.
333        ///
334        /// This avoids the problem of transitive chains that criss-cross crate
335        /// boundaries: for all local types `T`, you may assume that no other
336        /// crate will add `impl`s that allow comparing `T < U`. In other words,
337        /// if other crates add `impl`s that allow building longer transitive
338        /// chains `U1 < ... < T < V1 < ...`, then all the types that appear to
339        /// the right of `T` must be types that the crate defining `T` already
340        /// knows about. This rules out transitive chains where downstream
341        /// crates can add new `impl`s that "stitch together" comparisons of
342        /// foreign types in ways that violate transitivity.
343        ///
344        /// Not having such foreign `impl`s also avoids forward compatibility
345        /// issues where one crate adding more `PartialOrd` implementations can
346        /// cause build failures in downstream crates.
347        ///
348        /// ## Corollaries
349        ///
350        /// The following corollaries follow from the above requirements:
351        ///
352        /// - irreflexivity of `<` and `>`: `!(a < a)`, `!(a > a)`
353        /// - transitivity of `>`: if `a > b` and `b > c` then `a > c`
354        /// - duality of `partial_cmp`: `partial_cmp(a, b) == partial_cmp(b,
355        ///   a).map(Ordering::reverse)`
356        ///
357        /// ## Strict and non-strict partial orders
358        ///
359        /// The `<` and `>` operators behave according to a *strict* partial
360        /// order. However, `<=` and `>=` do **not** behave according to a
361        /// *non-strict* partial order. That is because mathematically, a
362        /// non-strict partial order would require reflexivity, i.e. `a <= a`
363        /// would need to be true for every `a`. This isn't always the case for
364        /// types that implement `PartialOrd`, for example:
365        ///
366        /// ```
367        /// let a = f64::sqrt(-1.0);
368        /// assert_eq!(a <= a, false);
369        /// ```
370        ///
371        /// ## How can I implement `PartialOrd`?
372        ///
373        /// `PartialOrd` only requires implementation of the [`PARTIAL_CMP`]
374        /// protocol, with the others generated from default implementations.
375        ///
376        /// However it remains possible to implement the others separately for
377        /// types which do not have a total order. For example, for floating
378        /// point numbers, `NaN < 0 == false` and `NaN >= 0 == false` (cf. IEEE
379        /// 754-2008 section 5.11).
380        ///
381        /// `PARTIAL_CMP` requires your type to be [`PARTIAL_EQ`].
382        ///
383        /// If your type is [`ORD`], you can implement [`PARTIAL_CMP`] by using
384        /// [`CMP`].
385        ///
386        /// You may also find it useful to use [`PARTIAL_CMP`] on your type's
387        /// fields.
388        ///
389        /// # Examples
390        ///
391        /// ```rune
392        /// let x = 0;
393        /// let y = 1;
394        ///
395        /// assert_eq!(x < y, true);
396        /// assert_eq!(x.lt(y), true);
397        /// ```
398        ///
399        /// [`partial_cmp`]: PartialOrd::partial_cmp
400        /// [`cmp`]: Ord::cmp
401    })?;

540    t.docs(docstring! {
541        /// Trait for types that form a [total
542        /// order](https://en.wikipedia.org/wiki/Total_order).
543        ///
544        /// Implementations must be consistent with the [`PartialOrd`]
545        /// implementation, and ensure `max`, `min`, and `clamp` are consistent
546        /// with `cmp`:
547        ///
548        /// - `partial_cmp(a, b) == Some(cmp(a, b))`.
549        /// - `max(a, b) == max_by(a, b, cmp)` (ensured by the default
550        ///   implementation).
551        /// - `min(a, b) == min_by(a, b, cmp)` (ensured by the default
552        ///   implementation).
553        /// - For `a.clamp(min, max)`, see the [method docs](#method.clamp)
554        ///   (ensured by the default implementation).
555        ///
556        /// It's easy to accidentally make `cmp` and `partial_cmp` disagree by
557        /// deriving some of the traits and manually implementing others.
558        ///
559        /// Violating these requirements is a logic error. The behavior
560        /// resulting from a logic error is not specified, but users of the
561        /// trait must ensure that such logic errors do *not* result in
562        /// undefined behavior. This means that `unsafe` code **must not** rely
563        /// on the correctness of these methods.
564        ///
565        /// ## Corollaries
566        ///
567        /// From the above and the requirements of `PartialOrd`, it follows that
568        /// for all `a`, `b` and `c`:
569        ///
570        /// - exactly one of `a < b`, `a == b` or `a > b` is true; and
571        /// - `<` is transitive: `a < b` and `b < c` implies `a < c`. The same
572        ///   must hold for both `==` and `>`.
573        ///
574        /// Mathematically speaking, the `<` operator defines a strict [weak
575        /// order]. In cases where `==` conforms to mathematical equality, it
576        /// also defines a strict [total order].
577        ///
578        /// [weak order]: https://en.wikipedia.org/wiki/Weak_ordering
579        /// [total order]: https://en.wikipedia.org/wiki/Total_order
580        ///
581        /// ## Lexicographical comparison
582        ///
583        /// Lexicographical comparison is an operation with the following
584        /// properties:
585        ///  - Two sequences are compared element by element.
586        ///  - The first mismatching element defines which sequence is
587        ///    lexicographically less or greater than the other.
588        ///  - If one sequence is a prefix of another, the shorter sequence is
589        ///    lexicographically less than the other.
590        ///  - If two sequences have equivalent elements and are of the same
591        ///    length, then the sequences are lexicographically equal.
592        ///  - An empty sequence is lexicographically less than any non-empty
593        ///    sequence.
594        ///  - Two empty sequences are lexicographically equal.
595        ///
596        /// ## How can I implement `Ord`?
597        ///
598        /// `Ord` requires that the type also be [`PARTIAL_RD`] and [`EQ`]
599        /// (which requires [`PARTIAL_EQ`]).
600        ///
601        /// Then you must define an implementation for [`CMP`]. You may find it
602        /// useful to use [`CMP`] on your type's fields.
603    })?;
604
605    t.handler(|cx| {
606        let cmp = cx.find(&Protocol::CMP)?;
607        let cmp = Caller::<(Value, Value), 2, Ordering>::new(cmp);
608
609        cx.find_or_define(&Protocol::MIN, {
610            let cmp = cmp.clone();
611
612            move |a: Value, b: Value| match vm_try!(cmp.call((a.clone(), b.clone()))) {
613                Ordering::Less | Ordering::Equal => VmResult::Ok(a),
614                Ordering::Greater => VmResult::Ok(b),
615            }
616        })?;
617
618        cx.find_or_define(&Protocol::MAX, {
619            let cmp = cmp.clone();
620
621            move |a: Value, b: Value| match vm_try!(cmp.call((a.clone(), b.clone()))) {
622                Ordering::Less | Ordering::Equal => VmResult::Ok(b),
623                Ordering::Greater => VmResult::Ok(a),
624            }
625        })?;
626
627        Ok(())
628    })?;
629
630    t.function("cmp")?
631        .argument_types::<(Value, Value)>()?
632        .return_type::<Ordering>()?
633        .docs(docstring! {
634            /// Compare two values.
635            ///
636            /// # Examples
637            ///
638            /// ```rune
639            /// use std::cmp::Ordering;
640            ///
641            /// assert_eq!(1.cmp(2), Ordering::Less);
642            /// assert_eq!(2.cmp(2), Ordering::Equal);
643            /// assert_eq!(2.cmp(1), Ordering::Greater);
644            /// ```
645        })?;
646
647    t.function("min")?
648        .argument_types::<(Value, Value)>()?
649        .return_type::<Ordering>()?
650        .docs(docstring! {
651            /// Return the minimum of two values.
652            ///
653            /// # Examples
654            ///
655            /// ```rune
656            /// assert_eq!(1.min(2), 1);
657            /// assert_eq!(2.min(2), 2);
658            /// assert_eq!(2.min(1), 1);
659            /// ```
660        })?;
661
662    t.function("max")?
663        .argument_types::<(Value, Value)>()?
664        .return_type::<Ordering>()?
665        .docs(docstring! {
666            /// Return the maximum of two values.
667            ///
668            /// # Examples
669            ///
670            /// ```rune
671            /// assert_eq!(1.max(2), 2);
672            /// assert_eq!(2.max(2), 2);
673            /// assert_eq!(2.max(1), 2);
674            /// ```
675        })?;
676
677    Ok(m)
678}

680/// Compares and returns the maximum of two values.
681///
682/// Returns the second argument if the comparison determines them to be equal.
683///
684/// Internally uses the [`CMP`] protocol.
685///
686/// # Examples
687///
688/// ```rune
689/// use std::cmp::max;
690///
691/// assert_eq!(max(1, 2), 2);
692/// assert_eq!(max(2, 2), 2);
693/// ```
694#[rune::function(keep)]
695fn max(v1: Value, v2: Value) -> VmResult<Value> {
696    VmResult::Ok(match vm_try!(Value::cmp(&v1, &v2)) {
697        Ordering::Less | Ordering::Equal => v2,
698        Ordering::Greater => v1,
699    })
700}
