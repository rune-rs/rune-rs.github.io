54/// Assemble a closure expression.
55#[instrument_ast(span = ast)]
56fn expr_call_closure<'hir>(
57    cx: &mut Ctxt<'hir, '_, '_>,
58    ast: &ast::ExprClosure,
59) -> compile::Result<hir::ExprKind<'hir>> {
60    alloc_with!(cx, ast);
61
62    let item =
63        cx.q.item_for("lowering closure call", ast.id)
64            .with_span(ast)?;
65
66    let Some(meta) = cx.q.query_meta(ast, item.item, Default::default())? else {
67        return Err(compile::Error::new(
68            ast,
69            ErrorKind::MissingItem {
70                item: cx.q.pool.item(item.item).try_to_owned()?,
71            },
72        ));
73    };
74
75    let meta::Kind::Closure { call, do_move, .. } = meta.kind else {
76        return Err(compile::Error::expected_meta(
77            ast,
78            meta.info(cx.q.pool)?,
79            "a closure",
80        ));
81    };
82
83    tracing::trace!("queuing closure build entry");
84
85    cx.scopes.push_captures()?;
86
87    let args = iter!(ast.args.as_slice(), |(arg, _)| fn_arg(cx, arg)?);
88    let body = alloc!(expr(cx, &ast.body)?);
89
90    let layer = cx.scopes.pop().with_span(&ast.body)?;
91
92    cx.q.set_used(&meta.item_meta)?;
93
94    let captures = &*iter!(layer.captures().map(|(_, id)| id));
95
96    let Some(queue) = cx.secondary_builds.as_mut() else {
97        return Err(compile::Error::new(ast, ErrorKind::ClosureInConst));
98    };
99
100    queue.try_push(SecondaryBuildEntry {
101        item_meta: meta.item_meta,
102        build: SecondaryBuild::Closure(Closure {
103            hir: alloc!(hir::ExprClosure {
104                args,
105                body,
106                captures,
107            }),
108            call,
109        }),
110    })?;
111
112    if captures.is_empty() {
113        return Ok(hir::ExprKind::Fn(meta.hash));
114    }
115
116    Ok(hir::ExprKind::CallClosure(alloc!(hir::ExprCallClosure {
117        hash: meta.hash,
118        do_move,
119        captures,
120    })))
121}

132#[instrument_ast(span = span)]
133fn statements<'hir>(
134    cx: &mut Ctxt<'hir, '_, '_>,
135    label: Option<&(ast::Label, T![:])>,
136    statements: &[ast::Stmt],
137    span: &dyn Spanned,
138) -> compile::Result<hir::Block<'hir>> {
139    alloc_with!(cx, span);
140
141    let label = match label {
142        Some((label, _)) => Some(alloc_str!(label.resolve(resolve_context!(cx.q))?)),
143        None => None,
144    };
145
146    cx.scopes.push(label)?;
147
148    let at = cx.statements.len();
149
150    let mut value = None;
151
152    for ast in statements {
153        let last = match ast {
154            ast::Stmt::Local(ast) => {
155                let depacked = if ast.attributes.is_empty() && cx.q.options.lowering > 0 {
156                    unpack_locals(cx, &ast.pat, &ast.expr)?
157                } else {
158                    false
159                };
160
161                if !depacked {
162                    let stmt = hir::Stmt::Local(alloc!(local(cx, ast)?));
163                    cx.statement_buffer.try_push(stmt)?;
164                }
165
166                value.take()
167            }
168            ast::Stmt::Expr(ast) => {
169                if let Some(stmt) = value.replace(&*alloc!(expr(cx, ast)?)).map(hir::Stmt::Expr) {
170                    cx.statement_buffer.try_push(stmt)?;
171                }
172
173                None
174            }
175            ast::Stmt::Semi(ast) => {
176                let stmt = hir::Stmt::Expr(alloc!(expr(cx, &ast.expr)?));
177                cx.statement_buffer.try_push(stmt)?;
178                value.take()
179            }
180            ast::Stmt::Item(..) => continue,
181        };
182
183        if let Some(last) = last {
184            cx.statements
185                .try_push(hir::Stmt::Expr(last))
186                .with_span(span)?;
187        }
188
189        for stmt in cx.statement_buffer.drain(..) {
190            cx.statements.try_push(stmt).with_span(span)?;
191        }
192    }
193
194    let statements = iter!(cx.statements.drain(at..));
195
196    let layer = cx.scopes.pop().with_span(span)?;
197
198    Ok(hir::Block {
199        span: span.span(),
200        label,
201        statements,
202        value,
203        drop: iter!(layer.into_drop_order()),
204    })
205}

207#[instrument_ast(span = ast)]
208fn expr_range<'hir>(
209    cx: &mut Ctxt<'hir, '_, '_>,
210    ast: &ast::ExprRange,
211) -> compile::Result<hir::ExprRange<'hir>> {
212    match (ast.start.as_deref(), ast.end.as_deref(), &ast.limits) {
213        (Some(start), None, ast::ExprRangeLimits::HalfOpen(..)) => Ok(hir::ExprRange::RangeFrom {
214            start: expr(cx, start)?,
215        }),
216        (None, None, ast::ExprRangeLimits::HalfOpen(..)) => Ok(hir::ExprRange::RangeFull),
217        (Some(start), Some(end), ast::ExprRangeLimits::Closed(..)) => {
218            Ok(hir::ExprRange::RangeInclusive {
219                start: expr(cx, start)?,
220                end: expr(cx, end)?,
221            })
222        }
223        (None, Some(end), ast::ExprRangeLimits::Closed(..)) => {
224            Ok(hir::ExprRange::RangeToInclusive {
225                end: expr(cx, end)?,
226            })
227        }
228        (None, Some(end), ast::ExprRangeLimits::HalfOpen(..)) => Ok(hir::ExprRange::RangeTo {
229            end: expr(cx, end)?,
230        }),
231        (Some(start), Some(end), ast::ExprRangeLimits::HalfOpen(..)) => Ok(hir::ExprRange::Range {
232            start: expr(cx, start)?,
233            end: expr(cx, end)?,
234        }),
235        (Some(..) | None, None, ast::ExprRangeLimits::Closed(..)) => Err(compile::Error::msg(
236            ast,
237            "Unsupported range, you probably want `..` instead of `..=`",
238        )),
239    }
240}

242#[instrument_ast(span = ast)]
243fn expr_object<'hir>(
244    cx: &mut Ctxt<'hir, '_, '_>,
245    ast: &ast::ExprObject,
246) -> compile::Result<hir::ExprKind<'hir>> {
247    alloc_with!(cx, ast);
248
249    let span = ast;
250    let mut keys_dup = HashMap::new();
251
252    let assignments = &mut *iter!(&ast.assignments, |(ast, _)| {
253        let key = object_key(cx, &ast.key)?;
254
255        if let Some(_existing) = keys_dup.try_insert(key.1, key.0)? {
256            return Err(compile::Error::new(
257                key.0,
258                ErrorKind::DuplicateObjectKey {
259                    #[cfg(feature = "emit")]
260                    existing: _existing.span(),
261                    #[cfg(feature = "emit")]
262                    object: key.0.span(),
263                },
264            ));
265        }
266
267        let assign = match &ast.assign {
268            Some((_, ast)) => expr(cx, ast)?,
269            None => {
270                let Some((name, _)) = cx.scopes.get(hir::Name::Str(key.1))? else {
271                    return Err(compile::Error::new(
272                        key.0,
273                        ErrorKind::MissingLocal {
274                            name: key.1.try_to_string()?.try_into()?,
275                        },
276                    ));
277                };
278
279                hir::Expr {
280                    span: ast.span(),
281                    kind: hir::ExprKind::Variable(name),
282                }
283            }
284        };
285
286        hir::FieldAssign {
287            key: (key.0.span(), key.1),
288            assign,
289            position: None,
290        }
291    });
292
293    let mut check_object_fields = |fields: &[meta::FieldMeta], item: &Item| {
294        let mut named = HashMap::new();
295
296        for f in fields.iter() {
297            named.try_insert(f.name.as_ref(), f)?;
298        }
299
300        for assign in assignments.iter_mut() {
301            match named.remove(assign.key.1) {
302                Some(field_meta) => {
303                    assign.position = Some(field_meta.position);
304                }
305                None => {
306                    return Err(compile::Error::new(
307                        assign.key.0,
308                        ErrorKind::LitObjectNotField {
309                            field: assign.key.1.try_into()?,
310                            item: item.try_to_owned()?,
311                        },
312                    ));
313                }
314            };
315        }
316
317        if let Some(field) = named.into_keys().next() {
318            return Err(compile::Error::new(
319                span,
320                ErrorKind::LitObjectMissingField {
321                    field: field.try_into()?,
322                    item: item.try_to_owned()?,
323                },
324            ));
325        }
326
327        Ok(())
328    };
329
330    let kind = match &ast.ident {
331        ast::ObjectIdent::Named(path) => {
332            let named = cx.q.convert_path(path)?;
333            let parameters = generics_parameters(cx, &named)?;
334            let meta = cx.lookup_meta(path, named.item, parameters)?;
335            let item = cx.q.pool.item(meta.item_meta.item);
336
337            match &meta.kind {
338                meta::Kind::Struct {
339                    fields: meta::Fields::Empty,
340                    constructor,
341                    ..
342                } => {
343                    check_object_fields(&[], item)?;
344
345                    match constructor {
346                        Some(_) => hir::ExprObjectKind::ExternalType {
347                            hash: meta.hash,
348                            args: 0,
349                        },
350                        None => hir::ExprObjectKind::Struct { hash: meta.hash },
351                    }
352                }
353                meta::Kind::Struct {
354                    fields: meta::Fields::Named(st),
355                    constructor,
356                    ..
357                } => {
358                    check_object_fields(&st.fields, item)?;
359
360                    match constructor {
361                        Some(_) => hir::ExprObjectKind::ExternalType {
362                            hash: meta.hash,
363                            args: st.fields.len(),
364                        },
365                        None => hir::ExprObjectKind::Struct { hash: meta.hash },
366                    }
367                }
368                _ => {
369                    return Err(compile::Error::new(
370                        span,
371                        ErrorKind::UnsupportedLitObject {
372                            meta: meta.info(cx.q.pool)?,
373                        },
374                    ));
375                }
376            }
377        }
378        ast::ObjectIdent::Anonymous(..) => hir::ExprObjectKind::Anonymous,
379    };
380
381    Ok(hir::ExprKind::Object(alloc!(hir::ExprObject {
382        kind,
383        assignments,
384    })))
385}

387/// Lower an expression.
388#[instrument_ast(span = ast)]
389pub(crate) fn expr<'hir>(
390    cx: &mut Ctxt<'hir, '_, '_>,
391    ast: &ast::Expr,
392) -> compile::Result<hir::Expr<'hir>> {
393    alloc_with!(cx, ast);
394
395    let in_path = take(&mut cx.in_path);
396
397    let kind = match ast {
398        ast::Expr::Path(ast) => expr_path(cx, ast, in_path)?,
399        ast::Expr::Assign(ast) => hir::ExprKind::Assign(alloc!(hir::ExprAssign {
400            lhs: expr(cx, &ast.lhs)?,
401            rhs: expr(cx, &ast.rhs)?,
402        })),
403        // TODO: lower all of these loop constructs to the same loop-like
404        // representation. We only do different ones here right now since it's
405        // easier when refactoring.
406        ast::Expr::While(ast) => {
407            let label = match &ast.label {
408                Some((label, _)) => Some(alloc_str!(label.resolve(resolve_context!(cx.q))?)),
409                None => None,
410            };
411
412            cx.scopes.push_loop(label)?;
413            let condition = condition(cx, &ast.condition)?;
414            let body = block(cx, None, &ast.body)?;
415            let layer = cx.scopes.pop().with_span(ast)?;
416
417            hir::ExprKind::Loop(alloc!(hir::ExprLoop {
418                label,
419                condition: Some(alloc!(condition)),
420                body,
421                drop: iter!(layer.into_drop_order()),
422            }))
423        }
424        ast::Expr::Loop(ast) => {
425            let label = match &ast.label {
426                Some((label, _)) => Some(alloc_str!(label.resolve(resolve_context!(cx.q))?)),
427                None => None,
428            };
429
430            cx.scopes.push_loop(label)?;
431            let body = block(cx, None, &ast.body)?;
432            let layer = cx.scopes.pop().with_span(ast)?;
433
434            let kind = hir::ExprKind::Loop(alloc!(hir::ExprLoop {
435                label,
436                condition: None,
437                body,
438                drop: iter!(layer.into_drop_order()),
439            }));
440
441            kind
442        }
443        ast::Expr::For(ast) => {
444            let iter = expr(cx, &ast.iter)?;
445
446            let label = match &ast.label {
447                Some((label, _)) => Some(alloc_str!(label.resolve(resolve_context!(cx.q))?)),
448                None => None,
449            };
450
451            cx.scopes.push_loop(label)?;
452            let binding = pat_binding(cx, &ast.binding)?;
453            let body = block(cx, None, &ast.body)?;
454
455            let layer = cx.scopes.pop().with_span(ast)?;
456
457            hir::ExprKind::For(alloc!(hir::ExprFor {
458                label,
459                binding,
460                iter,
461                body,
462                drop: iter!(layer.into_drop_order()),
463            }))
464        }
465        ast::Expr::Let(ast) => hir::ExprKind::Let(alloc!(hir::ExprLet {
466            pat: pat_binding(cx, &ast.pat)?,
467            expr: expr(cx, &ast.expr)?,
468        })),
469        ast::Expr::If(ast) => hir::ExprKind::If(alloc!(expr_if(cx, ast)?)),
470        ast::Expr::Match(ast) => hir::ExprKind::Match(alloc!(hir::ExprMatch {
471            expr: alloc!(expr(cx, &ast.expr)?),
472            branches: iter!(&ast.branches, |(ast, _)| {
473                cx.scopes.push(None)?;
474
475                let pat = pat_binding(cx, &ast.pat)?;
476                let condition = option!(&ast.condition, |(_, ast)| expr(cx, ast)?);
477                let body = expr(cx, &ast.body)?;
478
479                let layer = cx.scopes.pop().with_span(ast)?;
480
481                hir::ExprMatchBranch {
482                    span: ast.span(),
483                    pat,
484                    condition,
485                    body,
486                    drop: iter!(layer.into_drop_order()),
487                }
488            }),
489        })),
490        ast::Expr::Call(ast) => hir::ExprKind::Call(alloc!(expr_call(cx, ast)?)),
491        ast::Expr::FieldAccess(ast) => {
492            hir::ExprKind::FieldAccess(alloc!(expr_field_access(cx, ast)?))
493        }
494        ast::Expr::Empty(ast) => {
495            // NB: restore in_path setting.
496            cx.in_path = in_path;
497            hir::ExprKind::Group(alloc!(expr(cx, &ast.expr)?))
498        }
499        ast::Expr::Binary(ast) => {
500            let rhs_needs = match &ast.op {
501                ast::BinOp::As(..) | ast::BinOp::Is(..) | ast::BinOp::IsNot(..) => Needs::Type,
502                _ => Needs::Value,
503            };
504
505            let lhs = expr(cx, &ast.lhs)?;
506
507            let needs = replace(&mut cx.needs, rhs_needs);
508            let rhs = expr(cx, &ast.rhs)?;
509            cx.needs = needs;
510
511            hir::ExprKind::Binary(alloc!(hir::ExprBinary {
512                lhs,
513                op: ast.op,
514                rhs,
515            }))
516        }
517        ast::Expr::Unary(ast) => expr_unary(cx, ast)?,
518        ast::Expr::Index(ast) => hir::ExprKind::Index(alloc!(hir::ExprIndex {
519            target: expr(cx, &ast.target)?,
520            index: expr(cx, &ast.index)?,
521        })),
522        ast::Expr::Block(ast) => expr_block(cx, ast)?,
523        ast::Expr::Break(ast) => hir::ExprKind::Break(alloc!(expr_break(cx, ast)?)),
524        ast::Expr::Continue(ast) => hir::ExprKind::Continue(alloc!(expr_continue(cx, ast)?)),
525        ast::Expr::Yield(ast) => hir::ExprKind::Yield(option!(&ast.expr, |ast| expr(cx, ast)?)),
526        ast::Expr::Return(ast) => hir::ExprKind::Return(option!(&ast.expr, |ast| expr(cx, ast)?)),
527        ast::Expr::Await(ast) => hir::ExprKind::Await(alloc!(expr(cx, &ast.expr)?)),
528        ast::Expr::Try(ast) => hir::ExprKind::Try(alloc!(expr(cx, &ast.expr)?)),
529        ast::Expr::Select(ast) => {
530            let mut default = None;
531            let mut branches = Vec::new();
532            let mut exprs = Vec::new();
533
534            for (ast, _) in &ast.branches {
535                match ast {
536                    ast::ExprSelectBranch::Pat(ast) => {
537                        cx.scopes.push(None)?;
538
539                        let pat = pat_binding(cx, &ast.pat)?;
540                        let body = expr(cx, &ast.body)?;
541
542                        let layer = cx.scopes.pop().with_span(&ast)?;
543
544                        exprs.try_push(expr(cx, &ast.expr)?).with_span(&ast.expr)?;
545
546                        branches.try_push(hir::ExprSelectBranch {
547                            pat,
548                            body,
549                            drop: iter!(layer.into_drop_order()),
550                        })?;
551                    }
552                    ast::ExprSelectBranch::Default(ast) => {
553                        if default.is_some() {
554                            return Err(compile::Error::new(
555                                ast,
556                                ErrorKind::SelectMultipleDefaults,
557                            ));
558                        }
559
560                        default = Some(alloc!(expr(cx, &ast.body)?));
561                    }
562                }
563            }
564
565            hir::ExprKind::Select(alloc!(hir::ExprSelect {
566                branches: iter!(branches),
567                exprs: iter!(exprs),
568                default: option!(default),
569            }))
570        }
571        ast::Expr::Closure(ast) => expr_call_closure(cx, ast)?,
572        ast::Expr::Lit(ast) => hir::ExprKind::Lit(lit(cx, &ast.lit)?),
573        ast::Expr::Object(ast) => expr_object(cx, ast)?,
574        ast::Expr::Tuple(ast) => hir::ExprKind::Tuple(alloc!(hir::ExprSeq {
575            items: iter!(&ast.items, |(ast, _)| expr(cx, ast)?),
576        })),
577        ast::Expr::Vec(ast) => hir::ExprKind::Vec(alloc!(hir::ExprSeq {
578            items: iter!(&ast.items, |(ast, _)| expr(cx, ast)?),
579        })),
580        ast::Expr::Range(ast) => hir::ExprKind::Range(alloc!(expr_range(cx, ast)?)),
581        ast::Expr::Group(ast) => hir::ExprKind::Group(alloc!(expr(cx, &ast.expr)?)),
582        ast::Expr::MacroCall(ast) => {
583            let Some(id) = ast.id else {
584                return Err(compile::Error::msg(ast, "missing expanded macro id"));
585            };
586
587            match cx.q.builtin_macro_for(id).with_span(ast)?.as_ref() {
588                query::BuiltInMacro::Template(ast) => {
589                    let old = replace(&mut cx.in_template, true);
590
591                    let result = hir::ExprKind::Template(alloc!(hir::BuiltInTemplate {
592                        span: ast.span,
593                        from_literal: ast.from_literal,
594                        exprs: iter!(&ast.exprs, |ast| expr(cx, ast)?),
595                    }));
596
597                    cx.in_template = old;
598                    result
599                }
600                query::BuiltInMacro::Format(ast) => {
601                    let spec = hir::BuiltInFormatSpec {
602                        fill: ast.fill,
603                        align: ast.align,
604                        width: ast.width,
605                        precision: ast.precision,
606                        flags: ast.flags,
607                        format_type: ast.format_type,
608                    };
609
610                    hir::ExprKind::Format(alloc!(hir::BuiltInFormat {
611                        spec,
612                        value: alloc!(expr(cx, &ast.value)?),
613                    }))
614                }
615                query::BuiltInMacro::File(ast) => hir::ExprKind::Lit(lit(cx, &ast.value)?),
616                query::BuiltInMacro::Line(ast) => hir::ExprKind::Lit(lit(cx, &ast.value)?),
617            }
618        }
619    };
620
621    Ok(hir::Expr {
622        span: ast.span(),
623        kind,
624    })
625}

627/// Construct a pattern from a constant value.
628#[instrument_ast(span = span)]
629fn pat_const_value<'hir>(
630    cx: &mut Ctxt<'hir, '_, '_>,
631    const_value: &ConstValue,
632    span: &dyn Spanned,
633) -> compile::Result<hir::Pat<'hir>> {
634    alloc_with!(cx, span);
635
636    let kind = 'kind: {
637        let lit = match const_value.as_kind() {
638            ConstValueKind::Inline(value) => match *value {
639                Inline::Unit => {
640                    break 'kind hir::PatKind::Sequence(alloc!(hir::PatSequence {
641                        kind: hir::PatSequenceKind::Sequence {
642                            hash: runtime::Tuple::HASH,
643                            count: 0,
644                            is_open: false,
645                        },
646                        items: &[],
647                    }));
648                }
649                Inline::Bool(b) => hir::Lit::Bool(b),
650                Inline::Char(ch) => hir::Lit::Char(ch),
651                Inline::Unsigned(integer) => hir::Lit::Unsigned(integer),
652                Inline::Signed(integer) => hir::Lit::Signed(integer),
653                _ => {
654                    return Err(compile::Error::msg(
655                        span,
656                        "Unsupported constant value in pattern",
657                    ))
658                }
659            },
660            ConstValueKind::String(string) => hir::Lit::Str(alloc_str!(string.as_ref())),
661            ConstValueKind::Bytes(bytes) => hir::Lit::ByteStr(alloc_bytes!(bytes.as_ref())),
662            ConstValueKind::Instance(instance) => match &**instance {
663                ConstInstance {
664                    hash: runtime::Vec::HASH,
665                    variant_hash: Hash::EMPTY,
666                    fields,
667                } => {
668                    let items = iter!(fields.iter(), fields.len(), |value| pat_const_value(
669                        cx, value, span
670                    )?);
671
672                    break 'kind hir::PatKind::Sequence(alloc!(hir::PatSequence {
673                        kind: hir::PatSequenceKind::Sequence {
674                            hash: runtime::Vec::HASH,
675                            count: items.len(),
676                            is_open: false,
677                        },
678                        items,
679                    }));
680                }
681                ConstInstance {
682                    hash: runtime::OwnedTuple::HASH,
683                    variant_hash: Hash::EMPTY,
684                    fields,
685                } => {
686                    let items = iter!(fields.iter(), fields.len(), |value| pat_const_value(
687                        cx, value, span
688                    )?);
689
690                    break 'kind hir::PatKind::Sequence(alloc!(hir::PatSequence {
691                        kind: hir::PatSequenceKind::Sequence {
692                            hash: runtime::Vec::HASH,
693                            count: items.len(),
694                            is_open: false,
695                        },
696                        items,
697                    }));
698                }
699                ConstInstance {
700                    hash: runtime::Object::HASH,
701                    variant_hash: Hash::EMPTY,
702                    fields,
703                } => {
704                    let bindings = iter!(fields.iter(), fields.len(), |value| {
705                        let (key, value) = value.as_pair().with_span(span)?;
706                        let key = key.as_string().with_span(span)?;
707                        let pat = alloc!(pat_const_value(cx, value, span)?);
708                        hir::Binding::Binding(span.span(), alloc_str!(key.as_ref()), pat)
709                    });
710
711                    break 'kind hir::PatKind::Object(alloc!(hir::PatObject {
712                        kind: hir::PatSequenceKind::Sequence {
713                            hash: runtime::Object::HASH,
714                            count: bindings.len(),
715                            is_open: false,
716                        },
717                        bindings,
718                    }));
719                }
720                _ => {
721                    return Err(compile::Error::msg(
722                        span,
723                        "Unsupported constant value in pattern",
724                    ));
725                }
726            },
727        };
728
729        hir::PatKind::Lit(alloc!(hir::Expr {
730            span: span.span(),
731            kind: hir::ExprKind::Lit(lit),
732        }))
733    };
734
735    Ok(hir::Pat {
736        span: span.span(),
737        kind,
738    })
739}

741#[instrument_ast(span = ast)]
742fn expr_if<'hir>(
743    cx: &mut Ctxt<'hir, '_, '_>,
744    ast: &ast::ExprIf,
745) -> compile::Result<hir::Conditional<'hir>> {
746    alloc_with!(cx, ast);
747
748    let length = 1 + ast.expr_else_ifs.len();
749
750    let then = [(
751        ast.if_.span().join(ast.block.span()),
752        &ast.condition,
753        &ast.block,
754    )]
755    .into_iter();
756
757    let else_ifs = ast
758        .expr_else_ifs
759        .iter()
760        .map(|ast| (ast.span(), &ast.condition, &ast.block));
761
762    let branches = iter!(then.chain(else_ifs), length, |(span, c, b)| {
763        cx.scopes.push(None)?;
764
765        let condition = condition(cx, c)?;
766        let block = block(cx, None, b)?;
767
768        let layer = cx.scopes.pop().with_span(ast)?;
769
770        let condition = &*alloc!(condition);
771        let drop = &*iter!(layer.into_drop_order());
772
773        hir::ConditionalBranch {
774            span,
775            condition,
776            block,
777            drop,
778        }
779    });
780
781    let fallback = match &ast.expr_else {
782        Some(ast) => Some(&*alloc!(block(cx, None, &ast.block)?)),
783        None => None,
784    };
785
786    Ok(hir::Conditional { branches, fallback })
787}

789#[instrument_ast(span = ast)]
790fn lit<'hir>(cx: &mut Ctxt<'hir, '_, '_>, ast: &ast::Lit) -> compile::Result<hir::Lit<'hir>> {
791    alloc_with!(cx, ast);
792
793    match ast {
794        ast::Lit::Bool(lit) => Ok(hir::Lit::Bool(lit.value)),
795        ast::Lit::Number(lit) => {
796            let n = lit.resolve(resolve_context!(cx.q))?;
797
798            match (n.value, n.suffix) {
799                (ast::NumberValue::Float(n), _) => Ok(hir::Lit::Float(n)),
800                (ast::NumberValue::Integer(int), Some(ast::NumberSuffix::Unsigned(_, size))) => {
801                    let Some(n) = int.to_u64() else {
802                        return Err(compile::Error::new(
803                            ast,
804                            ErrorKind::BadUnsignedOutOfBounds { size },
805                        ));
806                    };
807
808                    if !size.unsigned_in(n) {
809                        return Err(compile::Error::new(
810                            ast,
811                            ErrorKind::BadUnsignedOutOfBounds { size },
812                        ));
813                    }
814
815                    Ok(hir::Lit::Unsigned(n))
816                }
817                (ast::NumberValue::Integer(int), Some(ast::NumberSuffix::Signed(_, size))) => {
818                    let Some(n) = int.to_i64() else {
819                        return Err(compile::Error::new(
820                            ast,
821                            ErrorKind::BadSignedOutOfBounds { size },
822                        ));
823                    };
824
825                    if !size.signed_in(n) {
826                        return Err(compile::Error::new(
827                            ast,
828                            ErrorKind::BadSignedOutOfBounds { size },
829                        ));
830                    }
831
832                    Ok(hir::Lit::Signed(n))
833                }
834                (ast::NumberValue::Integer(int), _) => {
835                    let Some(n) = int.to_i64() else {
836                        return Err(compile::Error::new(
837                            ast,
838                            ErrorKind::BadSignedOutOfBounds {
839                                size: NumberSize::S64,
840                            },
841                        ));
842                    };
843
844                    Ok(hir::Lit::Signed(n))
845                }
846            }
847        }
848        ast::Lit::Byte(lit) => {
849            let b = lit.resolve(resolve_context!(cx.q))?;
850            Ok(hir::Lit::Unsigned(b as u64))
851        }
852        ast::Lit::Char(lit) => {
853            let ch = lit.resolve(resolve_context!(cx.q))?;
854            Ok(hir::Lit::Char(ch))
855        }
856        ast::Lit::Str(lit) => {
857            let string = if cx.in_template {
858                lit.resolve_template_string(resolve_context!(cx.q))?
859            } else {
860                lit.resolve_string(resolve_context!(cx.q))?
861            };
862
863            Ok(hir::Lit::Str(alloc_str!(string.as_ref())))
864        }
865        ast::Lit::ByteStr(lit) => {
866            let bytes = lit.resolve(resolve_context!(cx.q))?;
867            Ok(hir::Lit::ByteStr(alloc_bytes!(bytes.as_ref())))
868        }
869    }
870}

875    ast: &ast::ExprUnary,
876) -> compile::Result<hir::ExprKind<'hir>> {
877    alloc_with!(cx, ast);
878
879    // NB: special unary expressions.
880    if let ast::UnOp::BorrowRef { .. } = ast.op {
881        return Err(compile::Error::new(ast, ErrorKind::UnsupportedRef));
882    }
883
884    let (
885        ast::UnOp::Neg(..),
886        ast::Expr::Lit(ast::ExprLit {

892        return Ok(hir::ExprKind::Unary(alloc!(hir::ExprUnary {
893            op: ast.op,
894            expr: expr(cx, &ast.expr)?,
895        })));
896    };
897
898    let number = n.resolve(resolve_context!(cx.q))?;
899
900    match (number.value, number.suffix) {
901        (ast::NumberValue::Float(n), _) => Ok(hir::ExprKind::Lit(hir::Lit::Float(-n))),
902        (ast::NumberValue::Integer(int), Some(ast::NumberSuffix::Unsigned(_, size))) => {
903            let Some(n) = int.neg().to_u64() else {
904                return Err(compile::Error::new(
905                    ast,
906                    ErrorKind::BadUnsignedOutOfBounds { size },
907                ));
908            };
909
910            if !size.unsigned_in(n) {
911                return Err(compile::Error::new(
912                    ast,
913                    ErrorKind::BadUnsignedOutOfBounds { size },
914                ));
915            }
916
917            Ok(hir::ExprKind::Lit(hir::Lit::Unsigned(n)))
918        }
919        (ast::NumberValue::Integer(int), Some(ast::NumberSuffix::Signed(_, size))) => {
920            let Some(n) = int.neg().to_i64() else {
921                return Err(compile::Error::new(
922                    ast,
923                    ErrorKind::BadSignedOutOfBounds { size },
924                ));
925            };
926
927            if !size.signed_in(n) {
928                return Err(compile::Error::new(
929                    ast,
930                    ErrorKind::BadSignedOutOfBounds { size },
931                ));
932            }
933
934            Ok(hir::ExprKind::Lit(hir::Lit::Signed(n)))
935        }
936        (ast::NumberValue::Integer(int), _) => {
937            let Some(n) = int.neg().to_i64() else {
938                return Err(compile::Error::new(
939                    ast,
940                    ErrorKind::BadSignedOutOfBounds {
941                        size: NumberSize::S64,
942                    },
943                ));
944            };
945
946            Ok(hir::ExprKind::Lit(hir::Lit::Signed(n)))
947        }
948    }

951/// Lower a block expression.
952#[instrument_ast(span = ast)]
953fn expr_block<'hir>(
954    cx: &mut Ctxt<'hir, '_, '_>,
955    ast: &ast::ExprBlock,
956) -> compile::Result<hir::ExprKind<'hir>> {
957    /// The kind of an [ExprBlock].
958    #[derive(Debug, Clone, Copy, PartialEq)]
959    #[non_exhaustive]
960    pub(crate) enum ExprBlockKind {
961        Default,
962        Async,
963        Const,
964    }
965
966    alloc_with!(cx, ast);
967
968    let kind = match (&ast.async_token, &ast.const_token) {
969        (Some(..), None) => ExprBlockKind::Async,
970        (None, Some(..)) => ExprBlockKind::Const,
971        _ => ExprBlockKind::Default,
972    };
973
974    if let ExprBlockKind::Default = kind {
975        return Ok(hir::ExprKind::Block(alloc!(block(
976            cx,
977            ast.label.as_ref(),
978            &ast.block
979        )?)));
980    }
981
982    if cx.const_eval {
983        // This only happens if the ast expression has not been indexed. Which
984        // only occurs during certain kinds of constant evaluation. So we limit
985        // evaluation to only support constant blocks.
986        let ExprBlockKind::Const = kind else {
987            return Err(compile::Error::msg(
988                ast,
989                "Only constant blocks are supported in this context",
990            ));
991        };
992
993        if let Some(label) = &ast.label {
994            return Err(compile::Error::msg(
995                label,
996                "Constant blocks cannot be labelled",
997            ));
998        };
999
1000        return Ok(hir::ExprKind::Block(alloc!(block(cx, None, &ast.block)?)));
1001    };
1002
1003    let item =
1004        cx.q.item_for("lowering block", ast.block.id)
1005            .with_span(&ast.block)?;
1006    let meta = cx.lookup_meta(ast, item.item, GenericsParameters::default())?;
1007
1008    match (kind, &meta.kind) {
1009        (ExprBlockKind::Async, &meta::Kind::AsyncBlock { call, do_move, .. }) => {
1010            tracing::trace!("queuing async block build entry");
1011
1012            if let Some(label) = &ast.label {
1013                return Err(compile::Error::msg(
1014                    label,
1015                    "Async blocks cannot be labelled",
1016                ));
1017            };
1018
1019            cx.scopes.push_captures()?;
1020            let block = alloc!(block(cx, None, &ast.block)?);
1021            let layer = cx.scopes.pop().with_span(&ast.block)?;
1022
1023            cx.q.set_used(&meta.item_meta)?;
1024
1025            let captures = &*iter!(layer.captures().map(|(_, id)| id));
1026
1027            let Some(queue) = cx.secondary_builds.as_mut() else {
1028                return Err(compile::Error::new(ast, ErrorKind::AsyncBlockInConst));
1029            };
1030
1031            queue.try_push(SecondaryBuildEntry {
1032                item_meta: meta.item_meta,
1033                build: SecondaryBuild::AsyncBlock(AsyncBlock {
1034                    hir: alloc!(hir::AsyncBlock { block, captures }),
1035                    call,
1036                }),
1037            })?;
1038
1039            Ok(hir::ExprKind::AsyncBlock(alloc!(hir::ExprAsyncBlock {
1040                hash: meta.hash,
1041                do_move,
1042                captures,
1043            })))
1044        }
1045        (ExprBlockKind::Const, meta::Kind::Const) => Ok(hir::ExprKind::Const(meta.hash)),
1046        _ => Err(compile::Error::expected_meta(
1047            ast,
1048            meta.info(cx.q.pool)?,
1049            "async or const block",
1050        )),
1051    }
1052}

1158/// The is a simple locals optimization which unpacks locals from a tuple and
1159/// assigns them directly to local.
1160fn unpack_locals(cx: &mut Ctxt<'_, '_, '_>, p: &ast::Pat, e: &ast::Expr) -> compile::Result<bool> {
1161    alloc_with!(cx, p);
1162
1163    match (p, e) {
1164        (p @ ast::Pat::Path(inner), e) => {
1165            let Some(ast::PathKind::Ident(..)) = inner.path.as_kind() else {
1166                return Ok(false);
1167            };
1168
1169            let e = expr(cx, e)?;
1170            let p = pat_binding(cx, p)?;
1171
1172            cx.statement_buffer
1173                .try_push(hir::Stmt::Local(alloc!(hir::Local {
1174                    span: p.span().join(e.span()),
1175                    pat: p,
1176                    expr: e,
1177                })))?;
1178
1179            return Ok(true);
1180        }
1181        (ast::Pat::Tuple(p), ast::Expr::Tuple(e)) => {
1182            if p.items.len() != e.items.len() {
1183                return Ok(false);
1184            }
1185
1186            for ((_, _), (p, _)) in e.items.iter().zip(&p.items) {
1187                if matches!(p, ast::Pat::Rest(..)) {
1188                    return Ok(false);
1189                }
1190            }
1191
1192            let mut exprs = Vec::new();
1193
1194            for (e, _) in &e.items {
1195                exprs.try_push(expr(cx, e)?)?;
1196            }
1197
1198            for (e, (p, _)) in exprs.into_iter().zip(&p.items) {
1199                let p = pat_binding(cx, p)?;
1200
1201                cx.statement_buffer
1202                    .try_push(hir::Stmt::Local(alloc!(hir::Local {
1203                        span: p.span().join(e.span()),
1204                        pat: p,
1205                        expr: e,
1206                    })))?;
1207            }
1208
1209            return Ok(true);
1210        }
1211        _ => {}
1212    };
1213
1214    Ok(false)
1215}

1229fn pat<'hir>(cx: &mut Ctxt<'hir, '_, '_>, ast: &ast::Pat) -> compile::Result<hir::Pat<'hir>> {
1230    fn filter((ast, _): &(ast::Pat, Option<ast::Comma>)) -> Option<&ast::Pat> {
1231        if matches!(ast, ast::Pat::Binding(..) | ast::Pat::Rest(..)) {
1232            return None;
1233        }
1234
1235        Some(ast)
1236    }
1237
1238    alloc_with!(cx, ast);
1239
1240    let kind = {
1241        match ast {
1242            ast::Pat::Ignore(..) => hir::PatKind::Ignore,
1243            ast::Pat::Path(ast) => {
1244                let named = cx.q.convert_path(&ast.path)?;
1245                let parameters = generics_parameters(cx, &named)?;
1246
1247                let path = 'path: {
1248                    if let Some(meta) = cx.try_lookup_meta(&ast, named.item, &parameters)? {
1249                        match meta.kind {
1250                            meta::Kind::Const => {
1251                                let Some(const_value) = cx.q.get_const_value(meta.hash) else {
1252                                    return Err(compile::Error::msg(
1253                                        ast,
1254                                        try_format!("Missing constant for hash {}", meta.hash),
1255                                    ));
1256                                };
1257
1258                                let const_value = const_value.try_clone().with_span(ast)?;
1259                                return pat_const_value(cx, &const_value, ast);
1260                            }
1261                            _ => {
1262                                if let Some((0, kind)) = tuple_match_for(&meta) {
1263                                    break 'path hir::PatPathKind::Kind(alloc!(kind));
1264                                }
1265                            }
1266                        }
1267                    };
1268
1269                    if let Some(ident) = ast.path.try_as_ident() {
1270                        let name = alloc_str!(ident.resolve(resolve_context!(cx.q))?);
1271                        let name = cx.scopes.define(hir::Name::Str(name), ast)?;
1272                        cx.pattern_bindings.try_push(name)?;
1273                        break 'path hir::PatPathKind::Ident(name);
1274                    }
1275
1276                    return Err(compile::Error::new(ast, ErrorKind::UnsupportedBinding));
1277                };
1278
1279                hir::PatKind::Path(alloc!(path))
1280            }
1281            ast::Pat::Lit(ast) => hir::PatKind::Lit(alloc!(expr(cx, &ast.expr)?)),
1282            ast::Pat::Vec(ast) => {
1283                let (is_open, count) = pat_items_count(ast.items.as_slice())?;
1284                let items = iter!(
1285                    ast.items.iter().filter_map(filter),
1286                    ast.items.len(),
1287                    |ast| pat(cx, ast)?
1288                );
1289
1290                hir::PatKind::Sequence(alloc!(hir::PatSequence {
1291                    kind: hir::PatSequenceKind::Sequence {
1292                        hash: runtime::Vec::HASH,
1293                        count,
1294                        is_open
1295                    },
1296                    items,
1297                }))
1298            }
1299            ast::Pat::Tuple(ast) => {
1300                let (is_open, count) = pat_items_count(ast.items.as_slice())?;
1301                let items = iter!(
1302                    ast.items.iter().filter_map(filter),
1303                    ast.items.len(),
1304                    |ast| pat(cx, ast)?
1305                );
1306
1307                let kind = if let Some(path) = &ast.path {
1308                    let named = cx.q.convert_path(path)?;
1309                    let parameters = generics_parameters(cx, &named)?;
1310                    let meta = cx.lookup_meta(path, named.item, parameters)?;
1311
1312                    // Treat the current meta as a tuple and get the number of arguments it
1313                    // should receive and the type check that applies to it.
1314                    let Some((args, kind)) = tuple_match_for(&meta) else {
1315                        return Err(compile::Error::expected_meta(
1316                            path,
1317                            meta.info(cx.q.pool)?,
1318                            "type that can be used in a tuple pattern",
1319                        ));
1320                    };
1321
1322                    if !(args == count || count < args && is_open) {
1323                        return Err(compile::Error::new(
1324                            path,
1325                            ErrorKind::BadArgumentCount {
1326                                expected: args,
1327                                actual: count,
1328                            },
1329                        ));
1330                    }
1331
1332                    kind
1333                } else {
1334                    hir::PatSequenceKind::Sequence {
1335                        hash: runtime::Tuple::HASH,
1336                        count,
1337                        is_open,
1338                    }
1339                };
1340
1341                hir::PatKind::Sequence(alloc!(hir::PatSequence { kind, items }))
1342            }
1343            ast::Pat::Object(ast) => {
1344                let (is_open, count) = pat_items_count(ast.items.as_slice())?;
1345
1346                let mut keys_dup = HashMap::new();
1347
1348                let bindings = iter!(ast.items.iter().take(count), |(pat, _)| {
1349                    let (key, binding) = match pat {
1350                        ast::Pat::Binding(binding) => {
1351                            let (span, key) = object_key(cx, &binding.key)?;
1352                            (
1353                                key,
1354                                hir::Binding::Binding(
1355                                    span.span(),
1356                                    key,
1357                                    alloc!(self::pat(cx, &binding.pat)?),
1358                                ),
1359                            )
1360                        }
1361                        ast::Pat::Path(path) => {
1362                            let Some(ident) = path.path.try_as_ident() else {
1363                                return Err(compile::Error::new(
1364                                    path,
1365                                    ErrorKind::UnsupportedPatternExpr,
1366                                ));
1367                            };
1368
1369                            let key = alloc_str!(ident.resolve(resolve_context!(cx.q))?);
1370                            let id = cx.scopes.define(hir::Name::Str(key), ident)?;
1371                            cx.pattern_bindings.try_push(id)?;
1372                            (key, hir::Binding::Ident(path.span(), key, id))
1373                        }
1374                        _ => {
1375                            return Err(compile::Error::new(
1376                                pat,
1377                                ErrorKind::UnsupportedPatternExpr,
1378                            ));
1379                        }
1380                    };
1381
1382                    if let Some(_existing) = keys_dup.try_insert(key, pat)? {
1383                        return Err(compile::Error::new(
1384                            pat,
1385                            ErrorKind::DuplicateObjectKey {
1386                                #[cfg(feature = "emit")]
1387                                existing: _existing.span(),
1388                                #[cfg(feature = "emit")]
1389                                object: pat.span(),
1390                            },
1391                        ));
1392                    }
1393
1394                    binding
1395                });
1396
1397                let kind = match &ast.ident {
1398                    ast::ObjectIdent::Named(path) => {
1399                        let named = cx.q.convert_path(path)?;
1400                        let parameters = generics_parameters(cx, &named)?;
1401                        let meta = cx.lookup_meta(path, named.item, parameters)?;
1402
1403                        let Some((mut fields, kind)) =
1404                            struct_match_for(&meta, is_open && count == 0)?
1405                        else {
1406                            return Err(compile::Error::expected_meta(
1407                                path,
1408                                meta.info(cx.q.pool)?,
1409                                "type that can be used in a struct pattern",
1410                            ));
1411                        };
1412
1413                        for binding in bindings.iter() {
1414                            if !fields.remove(binding.key()) {
1415                                return Err(compile::Error::new(
1416                                    ast,
1417                                    ErrorKind::LitObjectNotField {
1418                                        field: binding.key().try_into()?,
1419                                        item: cx.q.pool.item(meta.item_meta.item).try_to_owned()?,
1420                                    },
1421                                ));
1422                            }
1423                        }
1424
1425                        if !is_open && !fields.is_empty() {
1426                            let mut fields = fields.into_iter().try_collect::<Box<[_]>>()?;
1427
1428                            fields.sort();
1429
1430                            return Err(compile::Error::new(
1431                                ast,
1432                                ErrorKind::PatternMissingFields {
1433                                    item: cx.q.pool.item(meta.item_meta.item).try_to_owned()?,
1434                                    #[cfg(feature = "emit")]
1435                                    fields,
1436                                },
1437                            ));
1438                        }
1439
1440                        kind
1441                    }
1442                    ast::ObjectIdent::Anonymous(..) => hir::PatSequenceKind::Sequence {
1443                        hash: runtime::Object::HASH,
1444                        count,
1445                        is_open,
1446                    },
1447                };
1448
1449                hir::PatKind::Object(alloc!(hir::PatObject { kind, bindings }))
1450            }
1451            _ => {
1452                return Err(compile::Error::new(ast, ErrorKind::UnsupportedPatternExpr));
1453            }
1454        }
1455    };
1456
1457    Ok(hir::Pat {
1458        span: ast.span(),
1459        kind,
1460    })
1461}

1485/// Lower the given path.
1486#[instrument_ast(span = ast)]
1487fn expr_path<'hir>(
1488    cx: &mut Ctxt<'hir, '_, '_>,
1489    ast: &ast::Path,
1490    in_path: bool,
1491) -> compile::Result<hir::ExprKind<'hir>> {
1492    alloc_with!(cx, ast);
1493
1494    if let Some(ast::PathKind::SelfValue) = ast.as_kind() {
1495        let Some((id, _)) = cx.scopes.get(hir::Name::SelfValue)? else {
1496            return Err(compile::Error::new(ast, ErrorKind::MissingSelf));
1497        };
1498
1499        return Ok(hir::ExprKind::Variable(id));
1500    }
1501
1502    if let Needs::Value = cx.needs {
1503        if let Some(name) = ast.try_as_ident() {
1504            let name = alloc_str!(name.resolve(resolve_context!(cx.q))?);
1505
1506            if let Some((name, _)) = cx.scopes.get(hir::Name::Str(name))? {
1507                return Ok(hir::ExprKind::Variable(name));
1508            }
1509        }
1510    }
1511
1512    // Caller has indicated that if they can't have a variable, they do indeed
1513    // want a path.
1514    if in_path {
1515        return Ok(hir::ExprKind::Path);
1516    }
1517
1518    let named = cx.q.convert_path(ast)?;
1519    let parameters = generics_parameters(cx, &named)?;
1520
1521    if let Some(meta) = cx.try_lookup_meta(ast, named.item, &parameters)? {
1522        return expr_path_meta(cx, &meta, ast);
1523    }
1524
1525    if let (Needs::Value, Some(local)) = (cx.needs, ast.try_as_ident()) {
1526        let local = local.resolve(resolve_context!(cx.q))?;
1527
1528        // light heuristics, treat it as a type error in case the first
1529        // character is uppercase.
1530        if !local.starts_with(char::is_uppercase) {
1531            return Err(compile::Error::new(
1532                ast,
1533                ErrorKind::MissingLocal {
1534                    name: Box::<str>::try_from(local)?,
1535                },
1536            ));
1537        }
1538    }
1539
1540    let kind = if !parameters.parameters.is_empty() {
1541        ErrorKind::MissingItemParameters {
1542            item: cx.q.pool.item(named.item).try_to_owned()?,
1543            parameters: parameters.parameters,
1544        }
1545    } else {
1546        ErrorKind::MissingItem {
1547            item: cx.q.pool.item(named.item).try_to_owned()?,
1548        }
1549    };
1550
1551    Err(compile::Error::new(ast, kind))
1552}

1554/// Compile an item.
1555#[instrument_ast(span = span)]
1556fn expr_path_meta<'hir>(
1557    cx: &mut Ctxt<'hir, '_, '_>,
1558    meta: &meta::Meta,
1559    span: &dyn Spanned,
1560) -> compile::Result<hir::ExprKind<'hir>> {
1561    alloc_with!(cx, span);
1562
1563    if let Needs::Value = cx.needs {
1564        match &meta.kind {
1565            meta::Kind::Struct {
1566                fields: meta::Fields::Empty,
1567                ..
1568            } => Ok(hir::ExprKind::Call(alloc!(hir::ExprCall {
1569                call: hir::Call::Meta { hash: meta.hash },
1570                args: &[],
1571            }))),
1572            meta::Kind::Struct {
1573                fields: meta::Fields::Unnamed(0),
1574                ..
1575            } => Ok(hir::ExprKind::Call(alloc!(hir::ExprCall {
1576                call: hir::Call::Meta { hash: meta.hash },
1577                args: &[],
1578            }))),
1579            meta::Kind::Struct {
1580                fields: meta::Fields::Unnamed(..),
1581                ..
1582            } => Ok(hir::ExprKind::Fn(meta.hash)),
1583            meta::Kind::Function { .. } => Ok(hir::ExprKind::Fn(meta.hash)),
1584            meta::Kind::Const => Ok(hir::ExprKind::Const(meta.hash)),
1585            meta::Kind::Struct { .. } | meta::Kind::Type { .. } | meta::Kind::Enum { .. } => {
1586                Ok(hir::ExprKind::Type(Type::new(meta.hash)))
1587            }
1588            _ => Err(compile::Error::expected_meta(
1589                span,
1590                meta.info(cx.q.pool)?,
1591                "something that can be used as a value",
1592            )),
1593        }
1594    } else {
1595        let Some(type_hash) = meta.type_hash_of() else {
1596            return Err(compile::Error::expected_meta(
1597                span,
1598                meta.info(cx.q.pool)?,
1599                "something that has a type",
1600            ));
1601        };
1602
1603        Ok(hir::ExprKind::Type(Type::new(type_hash)))
1604    }
1605}

1622/// Test if the given pattern is open or not.
1623fn pat_items_count(items: &[(ast::Pat, Option<ast::Comma>)]) -> compile::Result<(bool, usize)> {
1624    let mut it = items.iter();
1625
1626    let (is_open, mut count) = match it.next_back() {
1627        Some((pat, _)) => matches!(pat, ast::Pat::Rest { .. })
1628            .then(|| (true, 0))
1629            .unwrap_or((false, 1)),
1630        None => return Ok((false, 0)),
1631    };
1632
1633    for (pat, _) in it {
1634        if let ast::Pat::Rest { .. } = pat {
1635            return Err(compile::Error::new(pat, ErrorKind::UnsupportedPatternRest));
1636        }
1637
1638        count += 1;
1639    }
1640
1641    Ok((is_open, count))
1642}

1644/// Generate a legal struct match for the given meta which indicates the type of
1645/// sequence and the fields that it expects.
1646///
1647/// For `open` matches (i.e. `{ .. }`), `Unnamed` and `Empty` structs are also
1648/// supported and they report empty fields.
1649fn struct_match_for(
1650    meta: &meta::Meta,
1651    open: bool,
1652) -> alloc::Result<Option<(HashSet<Box<str>>, hir::PatSequenceKind)>> {
1653    let (fields, kind) = match meta.kind {
1654        meta::Kind::Struct {
1655            ref fields,
1656            enum_hash,
1657            ..
1658        } => {
1659            let kind = 'kind: {
1660                if enum_hash != Hash::EMPTY {
1661                    break 'kind hir::PatSequenceKind::Type {
1662                        hash: enum_hash,
1663                        variant_hash: meta.hash,
1664                    };
1665                }
1666
1667                hir::PatSequenceKind::Type {
1668                    hash: meta.hash,
1669                    variant_hash: Hash::EMPTY,
1670                }
1671            };
1672
1673            (fields, kind)
1674        }
1675        meta::Kind::Type { .. } if open => {
1676            return Ok(Some((
1677                HashSet::new(),
1678                hir::PatSequenceKind::Type {
1679                    hash: meta.hash,
1680                    variant_hash: Hash::EMPTY,
1681                },
1682            )));
1683        }
1684        _ => {
1685            return Ok(None);
1686        }
1687    };
1688
1689    let fields = match fields {
1690        meta::Fields::Named(st) => st
1691            .fields
1692            .iter()
1693            .map(|f| f.name.try_clone())
1694            .try_collect::<alloc::Result<_>>()??,
1695        _ if open => HashSet::new(),
1696        _ => return Ok(None),
1697    };
1698
1699    Ok(Some((fields, kind)))
1700}

1702fn tuple_match_for(meta: &meta::Meta) -> Option<(usize, hir::PatSequenceKind)> {
1703    match meta.kind {
1704        meta::Kind::Struct {
1705            ref fields,
1706            enum_hash,
1707            ..
1708        } => {
1709            let args = match *fields {
1710                meta::Fields::Unnamed(args) => args,
1711                meta::Fields::Empty => 0,
1712                _ => return None,
1713            };
1714
1715            let kind = 'kind: {
1716                if enum_hash != Hash::EMPTY {
1717                    break 'kind hir::PatSequenceKind::Type {
1718                        hash: enum_hash,
1719                        variant_hash: meta.hash,
1720                    };
1721                }
1722
1723                hir::PatSequenceKind::Type {
1724                    hash: meta.hash,
1725                    variant_hash: Hash::EMPTY,
1726                }
1727            };
1728
1729            Some((args, kind))
1730        }
1731        _ => None,
1732    }
1733}

1735fn generics_parameters(
1736    cx: &mut Ctxt<'_, '_, '_>,
1737    named: &Named<'_>,
1738) -> compile::Result<GenericsParameters> {
1739    let mut parameters = GenericsParameters {
1740        trailing: named.trailing,
1741        parameters: [None, None],
1742    };
1743
1744    for (value, o) in named
1745        .parameters
1746        .iter()
1747        .zip(parameters.parameters.iter_mut())
1748    {
1749        if let &Some((span, generics)) = value {
1750            let mut builder = ParametersBuilder::new();
1751
1752            for (s, _) in generics {
1753                let hir::ExprKind::Type(ty) = expr(cx, &s.expr)?.kind else {
1754                    return Err(compile::Error::new(s, ErrorKind::UnsupportedGenerics));
1755                };
1756
1757                builder = builder.add(ty.into_hash()).with_span(span)?;
1758            }
1759
1760            *o = Some(builder.finish());
1761        }
1762    }
1763
1764    Ok(parameters)
1765}

1767/// Convert into a call expression.
1768#[instrument_ast(span = ast)]
1769fn expr_call<'hir>(
1770    cx: &mut Ctxt<'hir, '_, '_>,
1771    ast: &ast::ExprCall,
1772) -> compile::Result<hir::ExprCall<'hir>> {
1773    fn find_path(ast: &ast::Expr) -> Option<&ast::Path> {
1774        let mut current = ast;
1775
1776        loop {
1777            match current {
1778                ast::Expr::Path(path) => return Some(path),
1779                ast::Expr::Empty(ast) => {
1780                    current = &*ast.expr;
1781                    continue;
1782                }
1783                _ => return None,
1784            }
1785        }
1786    }
1787
1788    alloc_with!(cx, ast);
1789
1790    let in_path = replace(&mut cx.in_path, true);
1791    let expr = expr(cx, &ast.expr)?;
1792    cx.in_path = in_path;
1793
1794    let call = 'ok: {
1795        match expr.kind {
1796            hir::ExprKind::Variable(name) => {
1797                break 'ok hir::Call::Var { name };
1798            }
1799            hir::ExprKind::Path => {
1800                let Some(path) = find_path(&ast.expr) else {
1801                    return Err(compile::Error::msg(&ast.expr, "Expected path"));
1802                };
1803
1804                let named = cx.q.convert_path(path)?;
1805                let parameters = generics_parameters(cx, &named)?;
1806
1807                let meta = cx.lookup_meta(path, named.item, parameters)?;
1808                debug_assert_eq!(meta.item_meta.item, named.item);
1809
1810                match &meta.kind {
1811                    meta::Kind::Struct {
1812                        fields: meta::Fields::Empty,
1813                        ..
1814                    } => {
1815                        if !ast.args.is_empty() {
1816                            return Err(compile::Error::new(
1817                                &ast.args,
1818                                ErrorKind::BadArgumentCount {
1819                                    expected: 0,
1820                                    actual: ast.args.len(),
1821                                },
1822                            ));
1823                        }
1824                    }
1825                    meta::Kind::Struct {
1826                        fields: meta::Fields::Unnamed(args),
1827                        ..
1828                    } => {
1829                        if *args != ast.args.len() {
1830                            return Err(compile::Error::new(
1831                                &ast.args,
1832                                ErrorKind::BadArgumentCount {
1833                                    expected: *args,
1834                                    actual: ast.args.len(),
1835                                },
1836                            ));
1837                        }
1838
1839                        if *args == 0 {
1840                            cx.q.diagnostics.remove_tuple_call_parens(
1841                                cx.source_id,
1842                                &ast.args,
1843                                path,
1844                                None,
1845                            )?;
1846                        }
1847                    }
1848                    meta::Kind::Function { .. } => {
1849                        if let Some(message) = cx.q.lookup_deprecation(meta.hash) {
1850                            cx.q.diagnostics.used_deprecated(
1851                                cx.source_id,
1852                                &expr.span,
1853                                None,
1854                                message.try_into()?,
1855                            )?;
1856                        };
1857                    }
1858                    meta::Kind::ConstFn => {
1859                        let from =
1860                            cx.q.item_for("lowering constant function", ast.id)
1861                                .with_span(ast)?;
1862
1863                        break 'ok hir::Call::ConstFn {
1864                            from_module: from.module,
1865                            from_item: from.item,
1866                            id: meta.item_meta.item,
1867                        };
1868                    }
1869                    _ => {
1870                        return Err(compile::Error::expected_meta(
1871                            ast,
1872                            meta.info(cx.q.pool)?,
1873                            "something that can be called as a function",
1874                        ));
1875                    }
1876                };
1877
1878                break 'ok hir::Call::Meta { hash: meta.hash };
1879            }
1880            hir::ExprKind::FieldAccess(&hir::ExprFieldAccess {
1881                expr_field,
1882                expr: target,
1883            }) => {
1884                let hash = match expr_field {
1885                    hir::ExprField::Index(index) => Hash::index(index),
1886                    hir::ExprField::Ident(ident) => {
1887                        cx.q.unit.insert_debug_ident(ident)?;
1888                        Hash::ident(ident)
1889                    }
1890                    hir::ExprField::IdentGenerics(ident, hash) => {
1891                        cx.q.unit.insert_debug_ident(ident)?;
1892                        Hash::ident(ident).with_function_parameters(hash)
1893                    }
1894                };
1895
1896                break 'ok hir::Call::Associated {
1897                    target: alloc!(target),
1898                    hash,
1899                };
1900            }
1901            _ => {}
1902        }
1903
1904        break 'ok hir::Call::Expr { expr: alloc!(expr) };
1905    };
1906
1907    Ok(hir::ExprCall {
1908        call,
1909        args: iter!(&ast.args, |(ast, _)| self::expr(cx, ast)?),
1910    })
1911}

1913#[instrument_ast(span = ast)]
1914fn expr_field_access<'hir>(
1915    cx: &mut Ctxt<'hir, '_, '_>,
1916    ast: &ast::ExprFieldAccess,
1917) -> compile::Result<hir::ExprFieldAccess<'hir>> {
1918    alloc_with!(cx, ast);
1919
1920    let expr_field = match &ast.expr_field {
1921        ast::ExprField::LitNumber(ast) => {
1922            let number = ast.resolve(resolve_context!(cx.q))?;
1923
1924            let Some(index) = number.as_tuple_index() else {
1925                return Err(compile::Error::new(
1926                    ast,
1927                    ErrorKind::UnsupportedTupleIndex { number },
1928                ));
1929            };
1930
1931            hir::ExprField::Index(index)
1932        }
1933        ast::ExprField::Path(ast) => {
1934            let Some((ident, generics)) = ast.try_as_ident_generics() else {
1935                return Err(compile::Error::new(ast, ErrorKind::BadFieldAccess));
1936            };
1937
1938            let ident = alloc_str!(ident.resolve(resolve_context!(cx.q))?);
1939
1940            match generics {
1941                Some(generics) => {
1942                    let mut builder = ParametersBuilder::new();
1943
1944                    for (s, _) in generics {
1945                        let hir::ExprKind::Type(ty) = expr(cx, &s.expr)?.kind else {
1946                            return Err(compile::Error::new(s, ErrorKind::UnsupportedGenerics));
1947                        };
1948
1949                        builder = builder.add(ty.into_hash()).with_span(s)?;
1950                    }
1951
1952                    hir::ExprField::IdentGenerics(ident, builder.finish())
1953                }
1954                None => hir::ExprField::Ident(ident),
1955            }
1956        }
1957    };
1958
1959    Ok(hir::ExprFieldAccess {
1960        expr: expr(cx, &ast.expr)?,
1961        expr_field,
1962    })
