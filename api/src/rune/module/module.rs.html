45/// A [`Module`] that is a collection of native functions and types.
46///
47/// Needs to be installed into a [Context][crate::compile::Context] using
48/// [Context::install][crate::compile::Context::install].

50pub struct Module {

52    names: HashSet<Name>,

152    /// Register a type. Registering a type is mandatory in order to register
153    /// instance functions using that type.
154    ///
155    /// This will allow the type to be used within scripts, using the item named
156    /// here.
157    ///
158    /// # Examples
159    ///
160    /// ```
161    /// use rune::{Any, Context, Module};
162    ///
163    /// #[derive(Any)]
164    /// struct MyBytes {
165    ///     queue: Vec<String>,
166    /// }
167    ///
168    /// impl MyBytes {
169    ///     #[rune::function]
170    ///     fn len(&self) -> usize {
171    ///         self.queue.len()
172    ///     }
173    /// }
174    ///
175    /// // Register `len` without registering a type.
176    /// let mut m = Module::default();
177    /// // Note: cannot do this until we have registered a type.
178    /// m.function_meta(MyBytes::len)?;
179    ///
180    /// let mut context = rune::Context::new();
181    /// assert!(context.install(m).is_err());
182    ///
183    /// // Register `len` properly.
184    /// let mut m = Module::default();
185    ///
186    /// m.ty::<MyBytes>()?;
187    /// m.function_meta(MyBytes::len)?;
188    ///
189    /// let mut context = Context::new();
190    /// assert!(context.install(m).is_ok());
191    /// # Ok::<_, rune::support::Error>(())
192    /// ```
193    pub fn ty<T>(&mut self) -> Result<TypeMut<'_, T>, ContextError>
194    where
195        T: ?Sized + TypeOf + Named + InstallWith,
196    {
197        if !self.names.try_insert(Name::Item(T::HASH))? {
198            return Err(ContextError::ConflictingType {
199                item: T::ITEM.try_to_owned()?,
200                type_info: T::type_info(),
201                hash: T::HASH,
202            });
203        }
204
205        let index = self.types.len();
206        self.types_hash.try_insert(T::HASH, index)?;
207
208        self.types.try_push(ModuleType {
209            item: T::ITEM.try_to_owned()?,
210            hash: T::HASH,
211            common: ModuleItemCommon {
212                docs: Docs::EMPTY,
213                deprecated: None,
214            },
215            type_parameters: T::PARAMETERS,
216            type_info: T::type_info(),
217            spec: None,
218            constructor: None,
219        })?;
220
221        T::install_with(self)?;
222
223        let ty = self.types.last_mut().unwrap();
224
225        Ok(TypeMut {
226            docs: &mut ty.common.docs,
227            #[cfg(feature = "doc")]
228            deprecated: &mut ty.common.deprecated,
229            spec: &mut ty.spec,
230            constructor: &mut ty.constructor,
231            item: &ty.item,
232            _marker: PhantomData,
233        })
234    }

236    /// Accessor to modify type metadata such as documentaiton, fields, variants.
237    pub fn type_meta<T>(&mut self) -> Result<TypeMut<'_, T>, ContextError>
238    where
239        T: ?Sized + TypeOf + Named,
240    {
241        let type_hash = T::HASH;
242
243        let Some(ty) = self.types_hash.get(&type_hash).map(|&i| &mut self.types[i]) else {
244            let full_name = T::display().try_to_string()?;
245
246            return Err(ContextError::MissingType {
247                item: ItemBuf::with_item(&[full_name])?,
248                type_info: T::type_info(),
249            });
250        };
251
252        Ok(TypeMut {
253            docs: &mut ty.common.docs,
254            #[cfg(feature = "doc")]
255            deprecated: &mut ty.common.deprecated,
256            spec: &mut ty.spec,
257            constructor: &mut ty.constructor,
258            item: &ty.item,
259            _marker: PhantomData,
260        })
261    }

265    where

267    {
268        let type_hash = T::HASH;

273            return Err(ContextError::MissingType {
274                item: ItemBuf::with_item(&[full_name])?,
275                type_info: T::type_info(),

289            return Err(ContextError::MissingVariant {
290                type_info: T::type_info(),
291                index,
292            });
293        };
294
295        Ok(VariantMut {
296            name: variant.name,
297            docs: &mut variant.docs,
298            fields: &mut variant.fields,
299            constructor: &mut variant.constructor,
300            _marker: PhantomData,

340    pub(super) fn insert_constant<N, V>(
341        &mut self,
342        name: N,
343        value: V,
344    ) -> Result<ItemMut<'_>, ContextError>
345    where
346        N: IntoComponent,
347        V: TypeHash + TypeOf + ToConstValue,
348    {
349        let item = self.item.join([name])?;
350        let hash = Hash::type_hash(&item);
351
352        let value = match value.to_const_value() {
353            Ok(value) => value,
354            Err(error) => {
355                return Err(ContextError::InvalidConstValue {
356                    item,
357                    error: Box::try_new(error)?,
358                })
359            }
360        };
361
362        if !self.names.try_insert(Name::Item(hash))? {
363            return Err(ContextError::ConflictingConstantName { item, hash });
364        }
365
366        self.items.try_push(ModuleItem {
367            item,
368            hash,
369            common: ModuleItemCommon {
370                docs: Docs::EMPTY,
371                deprecated: None,
372            },
373            kind: ModuleItemKind::Constant(value),
374        })?;
375
376        self.insert_const_construct::<V>()?;
377
378        let c = self.items.last_mut().unwrap();
379
380        Ok(ItemMut {
381            docs: &mut c.common.docs,
382            #[cfg(feature = "doc")]
383            deprecated: &mut c.common.deprecated,
384        })
385    }

387    pub(super) fn insert_associated_constant<V>(
388        &mut self,
389        associated: Associated,
390        value: V,
391    ) -> Result<ItemMut<'_>, ContextError>
392    where
393        V: TypeHash + TypeOf + ToConstValue,
394    {
395        let value = match value.to_const_value() {
396            Ok(value) => value,
397            Err(error) => {
398                return Err(ContextError::InvalidAssociatedConstValue {
399                    container: associated.container_type_info,
400                    kind: Box::try_new(associated.name.kind)?,
401                    error: Box::try_new(error)?,
402                });
403            }
404        };
405
406        self.insert_associated_name(&associated)?;
407
408        self.associated.try_push(ModuleAssociated {
409            container: associated.container,
410            container_type_info: associated.container_type_info,
411            name: associated.name,
412            common: ModuleItemCommon {
413                docs: Docs::EMPTY,
414                deprecated: None,
415            },
416            kind: ModuleAssociatedKind::Constant(value),
417        })?;
418
419        self.insert_const_construct::<V>()?;
420
421        let last = self.associated.last_mut().unwrap();
422
423        Ok(ItemMut {
424            docs: &mut last.common.docs,
425            #[cfg(feature = "doc")]
426            deprecated: &mut last.common.deprecated,
427        })
428    }

430    fn insert_const_construct<V>(&mut self) -> alloc::Result<()>
431    where
432        V: TypeHash + TypeOf + ToConstValue,
433    {
434        if self.construct_hash.try_insert(V::HASH)? {
435            if let Some(construct) = V::construct()? {
436                self.construct
437                    .try_push((V::HASH, V::STATIC_TYPE_INFO, construct))?;
438            }
439        }
440
441        Ok(())
442    }

444    /// Register a native macro handler through its meta.
445    ///
446    /// The metadata must be provided by annotating the function with
447    /// [`#[rune::macro_]`][crate::macro_].
448    ///
449    /// This has the benefit that it captures documentation comments which can
450    /// be used when generating documentation or referencing the function
451    /// through code sense systems.
452    ///
453    /// # Examples
454    ///
455    /// ```
456    /// use rune::Module;
457    /// use rune::ast;
458    /// use rune::compile;
459    /// use rune::macros::{quote, MacroContext, TokenStream};
460    /// use rune::parse::Parser;
461    /// use rune::alloc::prelude::*;
462    ///
463    /// /// Takes an identifier and converts it into a string.
464    /// ///
465    /// /// # Examples
466    /// ///
467    /// /// ```rune
468    /// /// assert_eq!(ident_to_string!(Hello), "Hello");
469    /// /// ```
470    /// #[rune::macro_]
471    /// fn ident_to_string(cx: &mut MacroContext<'_, '_, '_>, stream: &TokenStream) -> compile::Result<TokenStream> {
472    ///     let mut p = Parser::from_token_stream(stream, cx.input_span());
473    ///     let ident = p.parse_all::<ast::Ident>()?;
474    ///     let ident = cx.resolve(ident)?.try_to_owned()?;
475    ///     let string = cx.lit(&ident)?;
476    ///     Ok(quote!(#string).into_token_stream(cx)?)
477    /// }
478    ///
479    /// let mut m = Module::new();
480    /// m.macro_meta(ident_to_string)?;
481    /// # Ok::<_, rune::support::Error>(())
482    /// ```
483    #[inline]
484    pub fn macro_meta(&mut self, meta: MacroMeta) -> Result<ItemMut<'_>, ContextError> {
485        let meta = meta()?;
486
487        let item = match meta.kind {
488            MacroMetaKind::Function(data) => {
489                let item = self.item.join(&data.item)?;
490                let hash = Hash::type_hash(&item);
491
492                if !self.names.try_insert(Name::Macro(hash))? {
493                    return Err(ContextError::ConflictingMacroName { item, hash });
494                }
495
496                let mut docs = Docs::EMPTY;
497                docs.set_docs(meta.docs)?;
498
499                self.items.try_push(ModuleItem {
500                    item,
501                    hash,
502                    common: ModuleItemCommon {
503                        docs,
504                        deprecated: None,
505                    },
506                    kind: ModuleItemKind::Macro(ModuleMacro {
507                        handler: data.handler,
508                    }),
509                })?;
510
511                self.items.last_mut().unwrap()
512            }
513            MacroMetaKind::Attribute(data) => {
514                let item = self.item.join(&data.item)?;
515                let hash = Hash::type_hash(&item);
516
517                if !self.names.try_insert(Name::AttributeMacro(hash))? {
518                    return Err(ContextError::ConflictingMacroName { item, hash });
519                }
520
521                let mut docs = Docs::EMPTY;
522                docs.set_docs(meta.docs)?;
523
524                self.items.try_push(ModuleItem {
525                    item,
526                    hash,
527                    common: ModuleItemCommon {
528                        docs,
529                        deprecated: None,
530                    },
531                    kind: ModuleItemKind::AttributeMacro(ModuleAttributeMacro {
532                        handler: data.handler,
533                    }),
534                })?;
535
536                self.items.last_mut().unwrap()
537            }
538        };
539
540        Ok(ItemMut {
541            docs: &mut item.common.docs,
542            #[cfg(feature = "doc")]
543            deprecated: &mut item.common.deprecated,
544        })
545    }

547    /// Register a native macro handler.
548    ///
549    /// If possible, [`Module::macro_meta`] should be used since it includes more
550    /// useful information about the macro.
551    ///
552    /// # Examples
553    ///
554    /// ```
555    /// use rune::Module;
556    /// use rune::ast;
557    /// use rune::compile;
558    /// use rune::macros::{quote, MacroContext, TokenStream};
559    /// use rune::parse::Parser;
560    /// use rune::alloc::prelude::*;
561    ///
562    /// fn ident_to_string(cx: &mut MacroContext<'_, '_, '_>, stream: &TokenStream) -> compile::Result<TokenStream> {
563    ///     let mut p = Parser::from_token_stream(stream, cx.input_span());
564    ///     let ident = p.parse_all::<ast::Ident>()?;
565    ///     let ident = cx.resolve(ident)?.try_to_owned()?;
566    ///     let string = cx.lit(&ident)?;
567    ///     Ok(quote!(#string).into_token_stream(cx)?)
568    /// }
569    ///
570    /// let mut m = Module::new();
571    /// m.macro_(["ident_to_string"], ident_to_string)?;
572    /// # Ok::<_, rune::support::Error>(())
573    /// ```
574    pub fn macro_<N, M>(&mut self, name: N, f: M) -> Result<ItemMut<'_>, ContextError>
575    where
576        M: 'static
577            + Send
578            + Sync
579            + Fn(&mut MacroContext<'_, '_, '_>, &TokenStream) -> compile::Result<TokenStream>,
580        N: IntoComponent,
581    {
582        let item = self.item.join([name])?;
583        let hash = Hash::type_hash(&item);
584
585        if !self.names.try_insert(Name::Macro(hash))? {
586            return Err(ContextError::ConflictingMacroName { item, hash });
587        }
588
589        let handler = MacroHandler::new(f)?;
590
591        self.items.try_push(ModuleItem {
592            item,
593            hash,
594            common: ModuleItemCommon::default(),
595            kind: ModuleItemKind::Macro(ModuleMacro { handler }),
596        })?;
597
598        let m = self.items.last_mut().unwrap();
599
600        Ok(ItemMut {
601            docs: &mut m.common.docs,
602            #[cfg(feature = "doc")]
603            deprecated: &mut m.common.deprecated,
604        })
605    }

675    ///
676    /// The metadata must be provided by annotating the function with
677    /// [`#[rune::function]`][macro@crate::function].
678    ///
679    /// This has the benefit that it captures documentation comments which can
680    /// be used when generating documentation or referencing the function
681    /// through code sense systems.
682    ///
683    /// # Examples
684    ///
685    /// ```
686    /// use rune::{ContextError, Module, Ref};
687    ///
688    /// /// This is a pretty neat function.
689    /// #[rune::function]
690    /// fn to_string(string: &str) -> String {
691    ///     string.to_string()
692    /// }
693    ///
694    /// /// This is a pretty neat download function
695    /// #[rune::function]
696    /// async fn download(url: Ref<str>) -> rune::support::Result<String> {
697    ///     # todo!()
698    /// }
699    ///
700    /// fn module() -> Result<Module, ContextError> {
701    ///     let mut m = Module::new();
702    ///     m.function_meta(to_string)?;
703    ///     m.function_meta(download)?;
704    ///     Ok(m)
705    /// }
706    /// ```
707    ///
708    /// Registering instance functions:
709    ///
710    /// ```
711    /// use rune::{Any, Module, Ref};
712    ///
713    /// #[derive(Any)]
714    /// struct MyBytes {
715    ///     queue: Vec<String>,
716    /// }
717    ///
718    /// impl MyBytes {
719    ///     fn new() -> Self {
720    ///         Self {
721    ///             queue: Vec::new(),
722    ///         }
723    ///     }
724    ///
725    ///     #[rune::function]
726    ///     fn len(&self) -> usize {
727    ///         self.queue.len()
728    ///     }
729    ///
730    ///     #[rune::function(instance, path = Self::download)]
731    ///     async fn download(this: Ref<Self>, url: Ref<str>) -> rune::support::Result<()> {
732    ///         # todo!()
733    ///     }
734    /// }
735    ///
736    /// let mut m = Module::default();
737    ///
738    /// m.ty::<MyBytes>()?;
739    /// m.function_meta(MyBytes::len)?;
740    /// m.function_meta(MyBytes::download)?;
741    /// # Ok::<_, rune::support::Error>(())
742    /// ```
743    #[inline]
744    pub fn function_meta(&mut self, meta: FunctionMeta) -> Result<ItemFnMut<'_>, ContextError> {
745        let meta = meta()?;
746
747        let mut docs = Docs::EMPTY;
748        docs.set_docs(meta.statics.docs)?;
749        docs.set_arguments(meta.statics.arguments)?;
750        let deprecated = meta.statics.deprecated.map(TryInto::try_into).transpose()?;
751
752        match meta.kind {
753            FunctionMetaKind::Function(data) => self.function_inner(data, docs, deprecated),
754            FunctionMetaKind::AssociatedFunction(data) => {
755                self.insert_associated_function(data, docs, deprecated)
756            }
757        }
758    }

773    ///
774    /// If possible, [`Module::function_meta`] should be used since it includes more
775    /// useful information about the function.
776    ///
777    /// # Examples
778    ///
779    /// ```
780    /// use rune::{docstring, Module};
781    ///
782    /// fn add_ten(value: i64) -> i64 {

791    ///         /// Adds 10 to any integer passed in.
792    ///     });
793    /// # Ok::<_, rune::support::Error>(())
794    /// ```
795    ///
796    /// Asynchronous function:
797    ///
798    /// ```
799    /// use rune::{docstring, Any, Module};
800    /// # async fn download(url: &str) -> Result<String, DownloadError> { Ok(String::new()) }
801    ///
802    /// #[derive(Any)]
803    /// struct DownloadError {
804    ///     /* .. */

807    /// async fn download_quote() -> Result<String, DownloadError> {

811    /// let mut module = Module::default();
812    ///
813    /// module.function("download_quote", download_quote)
814    ///     .build()?
815    ///     .docs(docstring! {
816    ///         /// Download a random quote from the internet.
817    ///     });
818    /// # Ok::<_, rune::support::Error>(())
819    /// ```
820    pub fn function<F, A, N, K>(&mut self, name: N, f: F) -> ModuleFunctionBuilder<'_, F, A, N, K>
821    where
822        F: Function<A, K, Return: MaybeTypeOf>,
823        A: FunctionArgs,
824        K: FunctionKind,
825    {
826        ModuleFunctionBuilder {
827            module: self,
828            inner: FunctionBuilder::new(name, f),
829        }
830    }

832    /// Register an instance function.
833    ///
834    /// If possible, [`Module::function_meta`] should be used since it includes
835    /// more useful information about the function.
836    ///
837    /// This returns a [`ItemMut`], which is a handle that can be used to
838    /// associate more metadata with the inserted item.
839    ///
840    /// # Replacing this with `function_meta` and `#[rune::function]`
841    ///
842    /// This is how you declare an instance function which takes `&self` or
843    /// `&mut self`:
844    ///
845    /// ```rust
846    /// # use rune::Any;
847    /// #[derive(Any)]
848    /// struct Struct {
849    ///     /* .. */
850    /// }
851    ///
852    /// impl Struct {
853    ///     /// Get the length of the `Struct`.
854    ///     #[rune::function]
855    ///     fn len(&self) -> usize {
856    ///         /* .. */
857    ///         # todo!()
858    ///     }
859    /// }
860    /// ```
861    ///
862    /// If a function does not take `&self` or `&mut self`, you must specify that
863    /// it's an instance function using `#[rune::function(instance)]`. The first
864    /// argument is then considered the instance the function gets associated with:
865    ///
866    /// ```rust
867    /// # use rune::Any;
868    /// #[derive(Any)]
869    /// struct Struct {
870    ///     /* .. */
871    /// }
872    ///
873    /// /// Get the length of the `Struct`.
874    /// #[rune::function(instance)]
875    /// fn len(this: &Struct) -> usize {
876    ///     /* .. */
877    ///     # todo!()
878    /// }
879    /// ```
880    ///
881    /// To declare an associated function which does not receive the type we
882    /// must specify the path to the function using `#[rune::function(path =
883    /// Self::<name>)]`:
884    ///
885    /// ```rust
886    /// # use rune::Any;
887    /// #[derive(Any)]
888    /// struct Struct {
889    ///     /* .. */
890    /// }
891    ///
892    /// impl Struct {
893    ///     /// Construct a new [`Struct`].
894    ///     #[rune::function(path = Self::new)]
895    ///     fn new() -> Struct {
896    ///         Struct {
897    ///            /* .. */
898    ///         }
899    ///     }
900    /// }
901    /// ```
902    ///
903    /// Or externally like this:
904    ///
905    /// ```rust
906    /// # use rune::Any;
907    /// #[derive(Any)]
908    /// struct Struct {
909    ///     /* .. */
910    /// }
911    ///
912    /// /// Construct a new [`Struct`].
913    /// #[rune::function(free, path = Struct::new)]
914    /// fn new() -> Struct {
915    ///     Struct {
916    ///        /* .. */
917    ///     }
918    /// }
919    /// ```
920    ///
921    /// The first part `Struct` in `Struct::new` is used to determine the type
922    /// the function is associated with.
923    ///
924    /// Protocol functions can either be defined in an impl block or externally.
925    /// To define a protocol externally, you can simply do this:
926    ///
927    /// ```rust
928    /// # use rune::Any;
929    /// # use rune::runtime::Formatter;
930    /// #[derive(Any)]
931    /// struct Struct {
932    ///     /* .. */
933    /// }
934    ///
935    /// #[rune::function(instance, protocol = DISPLAY_FMT)]
936    /// fn display_fmt(this: &Struct, f: &mut Formatter) -> std::fmt::Result {
937    ///     /* .. */
938    ///     # todo!()
939    /// }
940    /// ```
941    ///
942    /// # Examples
943    ///
944    /// ```
945    /// use rune::{Any, Module};
946    ///
947    /// #[derive(Any)]
948    /// struct MyBytes {
949    ///     queue: Vec<String>,
950    /// }
951    ///
952    /// impl MyBytes {
953    ///     /// Construct a new empty bytes container.
954    ///     #[rune::function(path = Self::new)]
955    ///     fn new() -> Self {
956    ///         Self {
957    ///             queue: Vec::new(),
958    ///         }
959    ///     }
960    ///
961    ///     /// Get the number of bytes.
962    ///     #[rune::function]
963    ///     fn len(&self) -> usize {
964    ///         self.queue.len()
965    ///     }
966    /// }
967    ///
968    /// let mut m = Module::default();
969    ///
970    /// m.ty::<MyBytes>()?;
971    /// m.function_meta(MyBytes::new)?;
972    /// m.function_meta(MyBytes::len)?;
973    /// # Ok::<_, rune::support::Error>(())
974    /// ```
975    ///
976    /// Asynchronous function:
977    ///
978    /// ```
979    /// use std::sync::atomic::AtomicU32;
980    /// use std::sync::Arc;
981    ///
982    /// use rune::{Any, Module, Ref};
983    ///
984    /// #[derive(Clone, Debug, Any)]
985    /// struct Client {
986    ///     value: Arc<AtomicU32>,
987    /// }
988    ///
989    /// #[derive(Any)]
990    /// struct DownloadError {
991    ///     /* .. */
992    /// }
993    ///
994    /// impl Client {
995    ///     /// Download a thing.
996    ///     #[rune::function(instance, path = Self::download)]
997    ///     async fn download(this: Ref<Self>) -> Result<(), DownloadError> {
998    ///         /* .. */
999    ///         # Ok(())
1000    ///     }
1001    /// }
1002    ///
1003    /// let mut module = Module::default();
1004    ///
1005    /// module.ty::<Client>()?;
1006    /// module.function_meta(Client::download)?;
1007    /// # Ok::<_, rune::support::Error>(())
1008    /// ```
1009    pub fn associated_function<N, F, A, K>(
1010        &mut self,
1011        name: N,
1012        f: F,
1013    ) -> Result<ItemFnMut<'_>, ContextError>
1014    where
1015        N: ToInstance,
1016        F: InstanceFunction<A, K, Return: MaybeTypeOf>,
1017        A: FunctionArgs,
1018        K: FunctionKind,
1019    {
1020        self.insert_associated_function(
1021            AssociatedFunctionData::from_instance_function(name.to_instance()?, f)?,
1022            Docs::EMPTY,
1023            None,
1024        )
1025    }

1119    fn function_inner(
1120        &mut self,
1121        data: FunctionData,
1122        docs: Docs,
1123        #[allow(unused)] deprecated: Option<Box<str>>,
1124    ) -> Result<ItemFnMut<'_>, ContextError> {
1125        let item = self.item.join(&data.item)?;
1126        let hash = Hash::type_hash(&item);
1127
1128        if !self.names.try_insert(Name::Item(hash))? {
1129            return Err(ContextError::ConflictingFunctionName { item, hash });
1130        }
1131
1132        self.items.try_push(ModuleItem {
1133            item,
1134            hash,
1135            common: ModuleItemCommon { docs, deprecated },
1136            kind: ModuleItemKind::Function(ModuleFunction {
1137                handler: data.handler,
1138                trait_hash: None,
1139                doc: DocFunction {
1140                    #[cfg(feature = "doc")]
1141                    is_async: data.is_async,
1142                    #[cfg(feature = "doc")]
1143                    args: data.args,
1144                    #[cfg(feature = "doc")]
1145                    return_type: data.return_type,
1146                    #[cfg(feature = "doc")]
1147                    argument_types: data.argument_types,
1148                },
1149            }),
1150        })?;
1151
1152        let last = self.items.last_mut().unwrap();
1153
1154        #[cfg(feature = "doc")]
1155        let last_fn = match &mut last.kind {
1156            ModuleItemKind::Function(f) => f,
1157            _ => unreachable!(),
1158        };
1159
1160        Ok(ItemFnMut {
1161            docs: &mut last.common.docs,
1162            #[cfg(feature = "doc")]
1163            deprecated: &mut last.common.deprecated,
1164            #[cfg(feature = "doc")]
1165            is_async: &mut last_fn.doc.is_async,
1166            #[cfg(feature = "doc")]
1167            args: &mut last_fn.doc.args,
1168            #[cfg(feature = "doc")]
1169            return_type: &mut last_fn.doc.return_type,
1170            #[cfg(feature = "doc")]
1171            argument_types: &mut last_fn.doc.argument_types,
1172        })
1173    }

1228    fn insert_associated_name(&mut self, associated: &Associated) -> Result<(), ContextError> {
1229        if !self
1230            .names
1231            .try_insert(Name::Associated(associated.as_key()?))?
1232        {
1233            return Err(match &associated.name.kind {
1234                meta::AssociatedKind::Protocol(protocol) => {
1235                    ContextError::ConflictingProtocolFunction {
1236                        type_info: associated.container_type_info.try_clone()?,
1237                        name: protocol.name.try_into()?,
1238                    }
1239                }
1240                meta::AssociatedKind::FieldFn(protocol, field) => {
1241                    ContextError::ConflictingFieldFunction {
1242                        type_info: associated.container_type_info.try_clone()?,
1243                        name: protocol.name.try_into()?,
1244                        field: field.as_ref().try_into()?,
1245                    }
1246                }
1247                meta::AssociatedKind::IndexFn(protocol, index) => {
1248                    ContextError::ConflictingIndexFunction {
1249                        type_info: associated.container_type_info.try_clone()?,
1250                        name: protocol.name.try_into()?,
1251                        index: *index,
1252                    }
1253                }
1254                meta::AssociatedKind::Instance(name) => ContextError::ConflictingInstanceFunction {
1255                    type_info: associated.container_type_info.try_clone()?,
1256                    name: name.as_ref().try_into()?,
1257                },
1258            });
1259        }
1260
1261        Ok(())
1262    }

1264    /// Define a new trait.
1265    pub fn define_trait(
1266        &mut self,
1267        item: impl IntoIterator<Item: IntoComponent>,
1268    ) -> Result<TraitMut<'_>, ContextError> {
1269        let item = self.item.join(item)?;
1270        let hash = Hash::type_hash(&item);
1271
1272        if !self.names.try_insert(Name::Item(hash))? {
1273            return Err(ContextError::ConflictingTrait { item, hash });
1274        }
1275
1276        self.traits.try_push(ModuleTrait {
1277            item,
1278            hash,
1279            common: ModuleItemCommon::default(),
1280            handler: None,
1281            functions: Vec::new(),
1282        })?;
1283
1284        let t = self.traits.last_mut().unwrap();
1285
1286        Ok(TraitMut {
1287            docs: &mut t.common.docs,
1288            #[cfg(feature = "doc")]
1289            deprecated: &mut t.common.deprecated,
1290            handler: &mut t.handler,
1291            functions: &mut t.functions,
1292        })
1293    }

1295    /// Implement the trait `trait_item` for the type `T`.
1296    pub fn implement_trait<T>(&mut self, trait_item: &Item) -> Result<(), ContextError>
1297    where
1298        T: ?Sized + TypeOf + Named,
1299    {
1300        let hash = T::HASH;
1301        let type_info = T::type_info();
1302        let trait_hash = Hash::type_hash(trait_item);
1303
1304        if !self.names.try_insert(Name::TraitImpl(hash, trait_hash))? {
1305            return Err(ContextError::ConflictingTraitImpl {
1306                trait_item: trait_item.try_to_owned()?,
1307                trait_hash,
1308                item: T::ITEM.try_to_owned()?,
1309                hash,
1310            });
1311        }
1312
1313        self.trait_impls.try_push(ModuleTraitImpl {
1314            item: T::ITEM.try_to_owned()?,
1315            hash,
1316            type_info,
1317            trait_item: trait_item.try_to_owned()?,
1318            trait_hash,
1319        })?;
1320
1321        Ok(())
1322    }
