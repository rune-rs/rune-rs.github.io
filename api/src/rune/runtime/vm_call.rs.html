11/// An instruction to push a virtual machine to the execution.
12#[derive(Debug)]
13#[must_use = "The construction of a vm call leaves the virtual machine stack in an intermediate state, VmCall::into_execution must be called to fix it"]
14pub(crate) struct VmCall {
15    /// The calling convention to use for the call.
16    call: Call,
17    /// Is set if the context differs for the call for the current virtual machine.
18    context: Option<Arc<RuntimeContext>>,
19    /// Is set if the unit differs for the call for the current virtual machine.
20    unit: Option<Arc<Unit>>,
21    /// The output to store the result of the call into.
22    out: Output,
23}
24
25impl VmCall {
26    pub(crate) fn new(
27        call: Call,
28        context: Option<Arc<RuntimeContext>>,
29        unit: Option<Arc<Unit>>,
30        out: Output,
31    ) -> Self {
32        Self {
33            call,
34            context,
35            unit,
36            out,
37        }
38    }
39
40    /// Encode the push itno an execution.
41    #[tracing::instrument(skip_all)]
42    pub(crate) fn into_execution<T>(self, execution: &mut VmExecution<T>) -> VmResult<()>
43    where
44        T: AsRef<Vm> + AsMut<Vm>,
45    {
46        let out = self.out;
47
48        let value = match self.call {
49            Call::Async => {
50                let vm = vm_try!(self.build_vm(execution));
51                let mut execution = vm.into_execution();
52                vm_try!(Value::try_from(vm_try!(Future::new(async move {
53                    execution.async_complete().await
54                }))))
55            }
56            Call::Immediate => {
57                vm_try!(execution.push_state(VmExecutionState {
58                    context: self.context,
59                    unit: self.unit,
60                }));
61
62                return VmResult::Ok(());
63            }
64            Call::Stream => {
65                let vm = vm_try!(self.build_vm(execution));
66                vm_try!(Value::try_from(Stream::new(vm)))
67            }
68            Call::Generator => {
69                let vm = vm_try!(self.build_vm(execution));
70                vm_try!(Value::try_from(Generator::new(vm)))
71            }
72        };
73
74        vm_try!(out.store(execution.vm_mut().stack_mut(), value));
75        VmResult::Ok(())
76    }
77
78    #[tracing::instrument(skip_all)]
79    fn build_vm<T>(self, execution: &mut VmExecution<T>) -> VmResult<Vm>
80    where
81        T: AsRef<Vm> + AsMut<Vm>,
82    {
83        let vm = execution.vm_mut();
84
85        let new_stack = vm_try!(vm.stack_mut().drain().try_collect::<Stack>());
86
87        let Some(ip) = vm.pop_call_frame_from_call() else {
88            return VmResult::err(VmErrorKind::MissingCallFrame);
89        };
90
91        let context = self.context.unwrap_or_else(|| vm.context().clone());
92        let unit = self.unit.unwrap_or_else(|| vm.unit().clone());
93
94        let mut vm = Vm::with_stack(context, unit, new_stack);
95        vm.set_ip(ip);
96        VmResult::Ok(vm)
97    }
98}
