22/// The type of a function in Rune.
23///
24/// Functions can be called using call expression syntax, such as `<expr>()`.
25///
26/// There are multiple different kind of things which can be coerced into a
27/// function in Rune:
28/// * Regular functions.
29/// * Closures (which might or might not capture their environment).
30/// * Built-in constructors for tuple types (tuple structs, tuple variants).
31///
32/// # Examples
33///
34/// ```rune
35/// // Captures the constructor for the `Some(<value>)` tuple variant.
36/// let build_some = Some;
37/// assert_eq!(build_some(42), Some(42));
38///
39/// fn build(value) {
40///     Some(value)
41/// }
42///
43/// // Captures the function previously defined.
44/// let build_some = build;
45/// assert_eq!(build_some(42), Some(42));
46/// ```
47#[derive(Any, TryClone)]
48#[repr(transparent)]
49#[rune(item = ::std::ops)]
50pub struct Function(FunctionImpl<Value>);

272    /// context which is not thread-safe.
273    ///
274    /// # Examples
275    ///
276    /// ```

290    /// let mut vm = Vm::without_runtime(Arc::new(unit));
291    /// let pony = vm.call(["main"], ())?;
292    /// let pony: Function = rune::from_value(pony)?;

312    ///         fn generator() {
313    ///             yield 42;
314    ///         }
315    ///
316    ///         pub fn main() {
317    ///             let g = generator();
318    ///
319    ///             move || {
320    ///                 g.next()
321    ///             }
322    ///         }
323    ///     }
324    /// };
325    ///
326    /// let unit = rune::prepare(&mut sources).build()?;
327    /// let mut vm = Vm::without_runtime(Arc::new(unit));
328    /// let closure = vm.call(["main"], ())?;
329    /// let closure: Function = rune::from_value(closure)?;
330    ///
331    /// // This is *not* fine since the returned closure has captured a
332    /// // generator which is not a constant value.
333    /// assert!(closure.into_sync().is_err());
334    /// # Ok::<_, rune::support::Error>(())
335    /// ```

512impl<V> FunctionImpl<V>
513where
514    V: TryClone,
515    OwnedTuple: TryFrom<Box<[V]>>,
516    VmErrorKind: From<<OwnedTuple as TryFrom<Box<[V]>>>::Error>,
517{
518    fn call<T>(&self, args: impl GuardedArgs) -> VmResult<T>
519    where
520        T: FromValue,
521    {
522        let value = match &self.inner {
523            Inner::FnHandler(handler) => {
524                let count = args.count();
525                let size = count.max(1);
526                // Ensure we have space for the return value.
527                let mut stack = vm_try!(Stack::with_capacity(size));
528                let _guard = vm_try!(unsafe { args.guarded_into_stack(&mut stack) });
529                vm_try!(stack.resize(size));
530                vm_try!((handler.handler)(
531                    &mut stack,
532                    InstAddress::ZERO,
533                    count,
534                    InstAddress::ZERO.output()
535                ));
536                stack.at(InstAddress::ZERO).clone()
537            }
538            Inner::FnOffset(fn_offset) => {
539                vm_try!(fn_offset.call(args, ()))
540            }
541            Inner::FnClosureOffset(closure) => {
542                let environment = vm_try!(closure.environment.try_clone());
543                let environment = vm_try!(OwnedTuple::try_from(environment));
544                vm_try!(closure.fn_offset.call(args, (environment,)))
545            }
546            Inner::FnUnitStruct(empty) => {
547                vm_try!(check_args(args.count(), 0));
548                vm_try!(Value::empty_struct(empty.rtti.clone()))
549            }
550            Inner::FnTupleStruct(tuple) => {
551                vm_try!(check_args(args.count(), tuple.args));
552                // SAFETY: We don't let the guard outlive the value.
553                let (args, _guard) = vm_try!(unsafe { args.guarded_into_vec() });
554                vm_try!(Value::tuple_struct(tuple.rtti.clone(), args))
555            }
556        };
557
558        VmResult::Ok(vm_try!(T::from_value(value)))
559    }
560
561    fn async_send_call<'a, A, T>(&'a self, args: A) -> impl Future<Output = VmResult<T>> + Send + 'a
562    where
563        A: 'a + Send + GuardedArgs,
564        T: 'a + Send + FromValue,
565    {
566        let future = async move {
567            let value: Value = vm_try!(self.call(args));
568
569            let value = match vm_try!(value.try_borrow_mut::<runtime::Future>()) {
570                Some(future) => vm_try!(future.await),
571                None => value,
572            };
573
574            VmResult::Ok(vm_try!(T::from_value(value)))
575        };
576
577        // Safety: Future is send because there is no way to call this
578        // function in a manner which allows any values from the future
579        // to escape outside of this future, hence it can only be
580        // scheduled by one thread at a time.
581        unsafe { AssertSend::new(future) }
582    }

584    /// Call with the given virtual machine. This allows for certain
585    /// optimizations, like avoiding the allocation of a new vm state in case
586    /// the call is internal.
587    ///
588    /// A stop reason will be returned in case the function call results in
589    /// a need to suspend the execution.
590    pub(crate) fn call_with_vm(
591        &self,
592        vm: &mut Vm,
593        addr: InstAddress,
594        args: usize,
595        out: Output,
596    ) -> VmResult<Option<VmHalt>> {
597        let reason = match &self.inner {
598            Inner::FnHandler(handler) => {
599                vm_try!((handler.handler)(vm.stack_mut(), addr, args, out));
600                None
601            }
602            Inner::FnOffset(fn_offset) => {
603                if let Some(vm_call) = vm_try!(fn_offset.call_with_vm(vm, addr, args, (), out)) {
604                    return VmResult::Ok(Some(VmHalt::VmCall(vm_call)));
605                }
606
607                None
608            }
609            Inner::FnClosureOffset(closure) => {
610                let environment = vm_try!(closure.environment.try_clone());
611                let environment = vm_try!(OwnedTuple::try_from(environment));
612
613                if let Some(vm_call) =
614                    vm_try!(closure
615                        .fn_offset
616                        .call_with_vm(vm, addr, args, (environment,), out))
617                {
618                    return VmResult::Ok(Some(VmHalt::VmCall(vm_call)));
619                }
620
621                None
622            }
623            Inner::FnUnitStruct(empty) => {
624                vm_try!(check_args(args, 0));
625                vm_try!(out.store(vm.stack_mut(), || Value::empty_struct(empty.rtti.clone())));
626                None
627            }
628            Inner::FnTupleStruct(tuple) => {
629                vm_try!(check_args(args, tuple.args));
630
631                let seq = vm_try!(vm.stack().slice_at(addr, args));
632                let data = seq.iter().cloned();
633                let value = vm_try!(AnySequence::new(tuple.rtti.clone(), data));
634                vm_try!(out.store(vm.stack_mut(), value));
635                None
636            }
637        };
638
639        VmResult::Ok(reason)
640    }

722impl FunctionImpl<Value> {
723    /// Try to convert into a [SyncFunction].
724    fn into_sync(self) -> Result<FunctionImpl<ConstValue>, RuntimeError> {
725        let inner = match self.inner {
726            Inner::FnClosureOffset(closure) => {
727                let mut env = Vec::try_with_capacity(closure.environment.len())?;
728
729                for value in Vec::from(closure.environment) {
730                    env.try_push(FromValue::from_value(value)?)?;
731                }
732
733                Inner::FnClosureOffset(FnClosureOffset {
734                    fn_offset: closure.fn_offset,
735                    environment: env.try_into_boxed_slice()?,
736                })
737            }
738            Inner::FnHandler(inner) => Inner::FnHandler(inner),
739            Inner::FnOffset(inner) => Inner::FnOffset(inner),
740            Inner::FnUnitStruct(inner) => Inner::FnUnitStruct(inner),
741            Inner::FnTupleStruct(inner) => Inner::FnTupleStruct(inner),
742        };
743
744        Ok(FunctionImpl { inner })
745    }
746}
747
748impl fmt::Debug for Function {
749    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
750        match &self.0.inner {
751            Inner::FnHandler(handler) => {
752                write!(f, "native function ({:p})", handler.handler.as_ref())?;
753            }
754            Inner::FnOffset(offset) => {
755                write!(f, "{} function (at: 0x{:x})", offset.call, offset.offset)?;
756            }
757            Inner::FnClosureOffset(closure) => {
758                write!(
759                    f,
760                    "closure (at: 0x{:x}, env:{:?})",
761                    closure.fn_offset.offset, closure.environment
762                )?;
763            }
764            Inner::FnUnitStruct(empty) => {
765                write!(f, "empty {}", empty.rtti.item)?;
766            }
767            Inner::FnTupleStruct(tuple) => {
768                write!(f, "tuple {}", tuple.rtti.item)?;
769            }
770        }
771
772        Ok(())
773    }
774}

776#[derive(Debug)]
777enum Inner<V> {
778    /// A native function handler.
779    /// This is wrapped as an `Arc<dyn FunctionHandler>`.
780    FnHandler(FnHandler),
781    /// The offset to a free function.
782    ///
783    /// This also captures the context and unit it belongs to allow for external
784    /// calls.
785    FnOffset(FnOffset),
786    /// A closure with a captured environment.
787    ///
788    /// This also captures the context and unit it belongs to allow for external
789    /// calls.
790    FnClosureOffset(FnClosureOffset<V>),
791    /// Constructor for a unit struct.
792    FnUnitStruct(FnUnitStruct),
793    /// Constructor for a tuple.
794    FnTupleStruct(FnTupleStruct),
795}
796
797impl<V> TryClone for Inner<V>
798where
799    V: TryClone,
800{
801    fn try_clone(&self) -> alloc::Result<Self> {
802        Ok(match self {
803            Inner::FnHandler(inner) => Inner::FnHandler(inner.clone()),
804            Inner::FnOffset(inner) => Inner::FnOffset(inner.clone()),
805            Inner::FnClosureOffset(inner) => Inner::FnClosureOffset(inner.try_clone()?),
806            Inner::FnUnitStruct(inner) => Inner::FnUnitStruct(inner.clone()),
807            Inner::FnTupleStruct(inner) => Inner::FnTupleStruct(inner.clone()),
808        })
809    }
810}

826#[derive(Clone, TryClone)]
827struct FnOffset {
828    context: Arc<RuntimeContext>,
829    /// The unit where the function resides.
830    unit: Arc<Unit>,
831    /// The offset of the function.
832    offset: usize,
833    /// The calling convention.
834    call: Call,
835    /// The number of arguments the function takes.
836    args: usize,
837    /// Hash for the function type
838    hash: Hash,
839}
840
841impl FnOffset {
842    /// Perform a call into the specified offset and return the produced value.
843    #[tracing::instrument(skip_all, fields(args = args.count(), extra = extra.count(), ?self.offset, ?self.call, ?self.args, ?self.hash))]
844    fn call(&self, args: impl GuardedArgs, extra: impl Args) -> VmResult<Value> {
845        vm_try!(check_args(
846            args.count().wrapping_add(extra.count()),
847            self.args
848        ));
849
850        let mut vm = Vm::new(self.context.clone(), self.unit.clone());
851
852        vm.set_ip(self.offset);
853        let _guard = vm_try!(unsafe { args.guarded_into_stack(vm.stack_mut()) });
854        vm_try!(extra.into_stack(vm.stack_mut()));
855
856        self.call.call_with_vm(vm)
857    }
858
859    /// Perform a potentially optimized call into the specified vm.
860    ///
861    /// This will cause a halt in case the vm being called into isn't the same
862    /// as the context and unit of the function.
863    #[tracing::instrument(skip_all, fields(args, extra = extra.count(), keep, ?self.offset, ?self.call, ?self.args, ?self.hash))]
864    fn call_with_vm(
865        &self,
866        vm: &mut Vm,
867        addr: InstAddress,
868        args: usize,
869        extra: impl Args,
870        out: Output,
871    ) -> VmResult<Option<VmCall>> {
872        vm_try!(check_args(args.wrapping_add(extra.count()), self.args));
873
874        let same_unit = matches!(self.call, Call::Immediate if vm.is_same_unit(&self.unit));
875        let same_context =
876            matches!(self.call, Call::Immediate if vm.is_same_context(&self.context));
877
878        vm_try!(vm.push_call_frame(self.offset, addr, args, Isolated::new(!same_context), out));
879        vm_try!(extra.into_stack(vm.stack_mut()));
880
881        // Fast path, just allocate a call frame and keep running.
882        if same_context && same_unit {
883            tracing::trace!("same context and unit");
884            return VmResult::Ok(None);
885        }
886
887        VmResult::Ok(Some(VmCall::new(
888            self.call,
889            (!same_context).then(|| self.context.clone()),
890            (!same_unit).then(|| self.unit.clone()),
891            out,
892        )))
893    }
894}
895
896impl fmt::Debug for FnOffset {
897    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
898        f.debug_struct("FnOffset")
899            .field("context", &(&self.context as *const _))
900            .field("unit", &(&self.unit as *const _))
901            .field("offset", &self.offset)
902            .field("call", &self.call)
903            .field("args", &self.args)
904            .finish()
905    }
906}

950fn check_args(actual: usize, expected: usize) -> VmResult<()> {
951    if actual != expected {
952        return VmResult::err(VmErrorKind::BadArgumentCount { expected, actual });
953    }

955    VmResult::Ok(())
