308    /// Set by index
309    ///
310    /// # Examples
311    ///
312    /// ```
313    /// use rune::runtime::Bytes;
314    ///
315    /// let mut bytes = Bytes::from_slice(b"abcd")?;
316    /// bytes.set(0, b'A');
317    /// assert_eq!(bytes, b"Abcd");
318    /// # Ok::<_, rune::support::Error>(())
319    /// ```
320    pub fn set(&mut self, index: usize, value: u8) -> VmResult<()> {
321        let Some(v) = self.bytes.get_mut(index) else {
322            return VmResult::err(VmErrorKind::OutOfRange {
323                index: index.into(),
324                length: self.len().into(),
325            });
326        };
327
328        *v = value;
329        VmResult::Ok(())
330    }

560/// This is a common index get implementation that is helpfull for custom type to impl `INDEX_GET` protocol.
561pub fn bytes_slice_index_get(this: &[u8], index: Value) -> VmResult<Option<Value>> {
562    let slice: Option<&[u8]> = 'out: {
563        if let Some(value) = index.as_any() {
564            match value.type_hash() {
565                RangeFrom::HASH => {
566                    let range = vm_try!(value.borrow_ref::<RangeFrom>());
567                    let start = vm_try!(range.start.as_usize());
568                    break 'out this.get(start..);
569                }
570                RangeFull::HASH => {
571                    _ = vm_try!(value.borrow_ref::<RangeFull>());
572                    break 'out this.get(..);
573                }
574                RangeInclusive::HASH => {
575                    let range = vm_try!(value.borrow_ref::<RangeInclusive>());
576                    let start = vm_try!(range.start.as_usize());
577                    let end = vm_try!(range.end.as_usize());
578                    break 'out this.get(start..=end);
579                }
580                RangeToInclusive::HASH => {
581                    let range = vm_try!(value.borrow_ref::<RangeToInclusive>());
582                    let end = vm_try!(range.end.as_usize());
583                    break 'out this.get(..=end);
584                }
585                RangeTo::HASH => {
586                    let range = vm_try!(value.borrow_ref::<RangeTo>());
587                    let end = vm_try!(range.end.as_usize());
588                    break 'out this.get(..end);
589                }
590                Range::HASH => {
591                    let range = vm_try!(value.borrow_ref::<Range>());
592                    let start = vm_try!(range.start.as_usize());
593                    let end = vm_try!(range.end.as_usize());
594                    break 'out this.get(start..end);
595                }
596                _ => {}
597            }
598        };
599
600        let index = vm_try!(usize::from_value(index));
601        let Some(value) = this.get(index) else {
602            return VmResult::Ok(None);
603        };
604
605        return VmResult::Ok(Some((*value).into()));
606    };
607
608    let Some(values) = slice else {
609        return VmResult::Ok(None);
610    };
611
612    let bytes = vm_try!(Bytes::try_from(values));
613    VmResult::Ok(Some(vm_try!(bytes.try_into())))
